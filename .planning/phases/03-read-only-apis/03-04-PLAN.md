        fires + paired-duration estimates (60s cap, FIFO per session)"). Algorithm:
          1. Detect the canonical hook event-name shape used by Claude Code OTEL
             (per Phase 2 RESEARCH §catalog: hooks emit pre/post pairs such as
             `claude_code.hook.pre_tool_use` ⇄ `claude_code.hook.post_tool_use`,
             `claude_code.hook.pre_compact` ⇄ `claude_code.hook.post_compact`,
             etc.). The pairing key is the substring AFTER `claude_code.hook.pre_`
             or `claude_code.hook.post_` (e.g. `tool_use`, `compact`).
          2. For each session_id, walk hook events ordered by ts ASC. Maintain a
             FIFO queue per pairing-key: when a `pre_*` event arrives, push its
             ts; when the matching `post_*` event arrives, pop the oldest pre-ts
             and emit a paired duration = (post.ts - pre.ts).total_seconds() * 1000.
          3. Cap each paired duration at 60_000 ms (drop higher; treat as
             abandoned hook per RESEARCH Pitfall — uncapped pairing produces
             tail-spike outliers from sessions left running overnight).
          4. Group paired durations by (local_day, hook_name) where hook_name is
             the canonical pre-event name (e.g. `claude_code.hook.pre_tool_use`).
             Compute p50 per group using Pattern 4 (offset percentile).
          5. Join the per-group p50 back into the per-day fires aggregation.
        If the canonical event-name catalog turns out to differ in production data
        (e.g., Phase 2 ingestor truncates the suffix), fall through gracefully:
        unpaired fires count toward `fires` but contribute nothing to
        `paired_duration_ms_p50` — the field is Optional[int].

      ```python
      _CACHE_TREND_SQL = text("""
          SELECT
            STRFTIME('%Y-%m-%d', day) AS day,
            COALESCE(SUM(tokens_cache_read), 0) AS cache_read,
            COALESCE(SUM(tokens_input), 0)      AS input,
            COALESCE(SUM(tokens_cache_create), 0) AS cache_create
          FROM token_usage
          WHERE day >= DATE('now', :since_clause, 'localtime')
          GROUP BY day
          ORDER BY day DESC
      """)

      @router.get("/usage/cache", response_model=CacheResponse)
      async def usage_cache(
          db: AsyncSession = Depends(get_session),
          range_: Literal["today", "7d", "30d"] = Query("7d", alias="range"),
      ):
          since = _RANGE_TO_SINCE[range_]
          rows = (await db.execute(_CACHE_TREND_SQL, {"since_clause": since})).mappings().all()
          trend: list[CacheTrendRow] = []
          tot_read = 0; tot_input = 0; tot_create = 0
          for r in rows:
              billable = (r["input"] or 0) + (r["cache_read"] or 0) + (r["cache_create"] or 0)
              hr = (r["cache_read"] or 0) / billable if billable else 0.0
              trend.append(CacheTrendRow(
                  day=r["day"],
                  hit_rate=round(hr, 4),
                  billable_tokens=billable,
                  low_sample=billable < 10_000,
              ))
              tot_read += r["cache_read"] or 0
              tot_input += r["input"] or 0
              tot_create += r["cache_create"] or 0
          tot_billable = tot_input + tot_read + tot_create
          overall_hr = round((tot_read / tot_billable) if tot_billable else 0.0, 4)
          return CacheResponse(
              hit_rate=overall_hr,
              trend=trend,
              range=RangeWindow(range_),
              low_sample=tot_billable < 10_000,
          )
      ```

    Step F — OBSV-05 /hooks/activity (fires + paired_duration_ms_p50, REQUIRED).

      Two SQL queries: (1) per-(day, hook_name) fires aggregation; (2) paired
      hook events for FIFO duration pairing. Pairing logic runs in Python over
      the second query's ordered output (cleaner than nested SQL window
      functions, well within RESEARCH §3 budget for Phase 3 reads).

      ```python
      from collections import defaultdict, deque

      _HOOK_FIRES_SQL = text("""
          SELECT
            STRFTIME('%Y-%m-%d', ts, 'localtime') AS day,
            event_name AS hook_name,
            COUNT(*) AS fires
          FROM otel_events
          WHERE event_name LIKE 'claude_code.hook%'
            AND ts >= datetime('now', :since_clause)
          GROUP BY day, event_name
          ORDER BY day DESC, fires DESC
      """)

      # Pull all hook events in window ordered by (session_id, ts) so the Python
      # pairing pass can FIFO-match pre→post within each session.
      _HOOK_EVENTS_SQL = text("""
          SELECT
            STRFTIME('%Y-%m-%d', ts, 'localtime') AS day,
            ts,
            event_name,
            session_id
          FROM otel_events
          WHERE event_name LIKE 'claude_code.hook%'
            AND ts >= datetime('now', :since_clause)
          ORDER BY session_id ASC, ts ASC
      """)

      _HOOK_PAIR_CAP_MS = 60_000

      def _classify_hook(event_name: str) -> tuple[str, str] | None:
          """Return (kind, key) where kind in {'pre','post'} and key is the suffix
          shared by the pair (e.g. 'tool_use'). None if not a paired hook event."""
          # Canonical Claude Code hook OTEL names: claude_code.hook.pre_<key> /
          # claude_code.hook.post_<key>. Fall through gracefully on other shapes.
          PRE = "claude_code.hook.pre_"
          POST = "claude_code.hook.post_"
          if event_name.startswith(PRE):
              return ("pre", event_name[len(PRE):])
          if event_name.startswith(POST):
              return ("post", event_name[len(POST):])
          return None

      def _percentile(sorted_values: list[int], p: float) -> int | None:
          if not sorted_values:
              return None
          # Pattern 4 offset (Pitfall 2 wrapper): max(int(N*p) - 1, 0)
          idx = max(int(len(sorted_values) * p) - 1, 0)
          return sorted_values[idx]

      @router.get("/hooks/activity", response_model=HookActivityResponse)
      async def hooks_activity(
          db: AsyncSession = Depends(get_session),
          range_: Literal["today", "7d", "30d"] = Query("7d", alias="range"),
      ):
          since = _RANGE_TO_SINCE[range_]
          fires_rows = (await db.execute(_HOOK_FIRES_SQL, {"since_clause": since})).mappings().all()
          ev_rows = (await db.execute(_HOOK_EVENTS_SQL, {"since_clause": since})).mappings().all()

          # FIFO pairing per session_id, per pairing-key.
          # Output: durations[(day, "claude_code.hook.pre_<key>")] = [ms, ms, ...]
          # We attribute each paired duration to the DAY of the PRE event and to
          # the canonical PRE event_name (consistent with the fires aggregation).
          per_session_queues: dict[tuple[str, str], deque] = defaultdict(deque)
          # key = (session_id, pairing_key); value = deque of (day, pre_ts)
          durations: dict[tuple[str, str], list[int]] = defaultdict(list)
          current_session: str | None = None
          for r in ev_rows:
              # Reset queue context when session changes (rows ordered by session)
              if r["session_id"] != current_session:
                  current_session = r["session_id"]
              cls = _classify_hook(r["event_name"])
              if cls is None or current_session is None:
                  continue
              kind, pair_key = cls
              q_key = (current_session, pair_key)
              if kind == "pre":
                  per_session_queues[q_key].append((r["day"], r["ts"]))
              else:  # post
                  if not per_session_queues[q_key]:
                      continue  # orphan post — drop
                  pre_day, pre_ts = per_session_queues[q_key].popleft()
                  # Compute duration. ts comes back as a string from SQLite text().
                  try:
                      pre_dt = datetime.fromisoformat(str(pre_ts))
                      post_dt = datetime.fromisoformat(str(r["ts"]))
                      if pre_dt.tzinfo is None:
                          pre_dt = pre_dt.replace(tzinfo=timezone.utc)
                      if post_dt.tzinfo is None:
                          post_dt = post_dt.replace(tzinfo=timezone.utc)
                      ms = int((post_dt - pre_dt).total_seconds() * 1000)
                  except (ValueError, TypeError):
                      continue
                  if ms < 0:
                      continue
                  ms = min(ms, _HOOK_PAIR_CAP_MS)
                  bucket_key = (pre_day, f"claude_code.hook.pre_{pair_key}")
                  durations[bucket_key].append(ms)

          # Compute p50 per bucket.
          p50_by_key: dict[tuple[str, str], int | None] = {}
          for k, vs in durations.items():
              vs.sort()
              p50_by_key[k] = _percentile(vs, 0.5)

          items: list[HookActivityRow] = []
          for r in fires_rows:
              key = (r["day"], r["hook_name"])
              items.append(HookActivityRow(
                  day=r["day"],
                  hook_name=r["hook_name"],
                  fires=int(r["fires"]),
                  paired_duration_ms_p50=p50_by_key.get(key),
              ))
          total = sum(r.fires for r in items)
          return HookActivityResponse(items=items, range=RangeWindow(range_), total_fires=total)
      ```
