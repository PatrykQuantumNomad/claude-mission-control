      ```python
      @router.get("/system/state", response_model=SystemStateResponse)
      async def system_state(
          key: Optional[str] = Query(None, description="If provided, return single key's value"),
          db: AsyncSession = Depends(get_session),
      ):
          if key is not None:
              if key not in _SYSTEM_STATE_WHITELIST:
                  raise HTTPException(status_code=404, detail=f"key not in whitelist")
              row = (await db.execute(
                  select(SystemState).where(SystemState.key == key)
              )).scalar_one_or_none()
              if row is None:
                  raise HTTPException(status_code=404, detail="key not found")
              return SystemStateResponse(items={key: _coerce_value(row)})
          # No specific key: return all whitelisted keys present in DB
          rows = (await db.execute(
              select(SystemState).where(SystemState.key.in_(_SYSTEM_STATE_WHITELIST))
          )).scalars().all()
          return SystemStateResponse(items={r.key: _coerce_value(r) for r in rows})

      def _coerce_value(row: SystemState):
          # Prefer value_json if set, else value (str). Same pattern other phases follow.
          if row.value_json is not None:
              return row.value_json
          return row.value
      ```

    Step D — Implement /attention (SAPI-04). Per RESEARCH Pitfall 7: read EVERY field;
    return 0/None for things Phase 4 hasn't wired:
      ```python
      @router.get("/attention", response_model=AttentionResponse)
      async def attention(db: AsyncSession = Depends(get_session)):
          # Phase-4-deferred fields. Per Pitfall 7 we DO NOT branch on table emptiness —
          # return 0 explicitly so the contract doesn't change when Phase 4 lands.
          pending_decisions = 0
          failed_tasks = 0

          # Stuck sessions: started > 3h ago and never ended.
          stuck_q = select(func.count()).select_from(SessionModel).where(
              SessionModel.ended_at.is_(None),
              SessionModel.started_at < func.datetime("now", "-3 hours"),
          )
          stuck_sessions = (await db.execute(stuck_q)).scalar_one()

          # Stale dispatcher: derive from system_state.dispatcher_last_tick_at age.
          stale_dispatcher_seconds: Optional[int] = None
          row = (await db.execute(
              select(SystemState).where(SystemState.key == "dispatcher_last_tick_at")
          )).scalar_one_or_none()
          if row and row.value:
              try:
                  ts = datetime.fromisoformat(row.value)
                  if ts.tzinfo is None:
                      ts = ts.replace(tzinfo=timezone.utc)
                  stale_dispatcher_seconds = int(
                      (datetime.now(timezone.utc) - ts).total_seconds()
                  )
              except ValueError:
                  pass

          items: list[AttentionItem] = []
          if stuck_sessions > 0:
              items.append(AttentionItem(
                  kind="stuck_sessions",
                  severity="warning",
                  count=stuck_sessions,
                  detail=f"{stuck_sessions} session(s) running >3h with no end timestamp",
              ))
          if stale_dispatcher_seconds is not None and stale_dispatcher_seconds > 600:
              items.append(AttentionItem(
                  kind="stale_dispatcher",
                  severity="error",
                  count=1,
                  detail=f"dispatcher last tick {stale_dispatcher_seconds}s ago",
              ))

          return AttentionResponse(
              items=items,
              pending_decisions=pending_decisions,
              failed_tasks=failed_tasks,
              stale_dispatcher_seconds=stale_dispatcher_seconds,
              stuck_sessions=stuck_sessions,
          )
      ```

      ```python
      from fastapi.sse import EventSourceResponse
      from cmc.api.sse import tail_otel_events

      @router.get("/firehose")
      async def firehose(
          request: Request,
          since: Optional[str] = Query(None, description="ISO timestamp; default = tail"),
          event_name: Optional[str] = Query(None, description="server-side event_name filter"),
          db: AsyncSession = Depends(get_session),
      ):
          """SSE stream of recent OTEL events (SAPI-05).

          Per RESEARCH Pitfall 1 + Pitfall 3: tail_otel_events handles disconnect
          detection, batched fetches, max-duration cap. We just convert `since` to
          a starting id and hand off.
          """
          since_id: Optional[int] = None
          if since:
              try:
                  ts = datetime.fromisoformat(since)
                  if ts.tzinfo is None:
                      ts = ts.replace(tzinfo=timezone.utc)
              except ValueError:
                  raise HTTPException(status_code=400, detail="invalid `since` ISO timestamp")
              row = (await db.execute(
                  select(func.max(OtelEvent.id)).where(OtelEvent.ts < ts)
              )).scalar_one_or_none()
              since_id = (row or 0)

          generator = tail_otel_events(
              request, db, since_id=since_id, event_name=event_name
          )
          return EventSourceResponse(generator)
      ```

      Note: the `db` AsyncSession dependency-injected here is held for the lifetime
      of the SSE response (FastAPI scopes deps to the response). The
      tail_otel_events helper's per-iteration query reuses this connection (RESEARCH
      Pitfall 3 — exhaust results to list before sleep, never hold cursors).
