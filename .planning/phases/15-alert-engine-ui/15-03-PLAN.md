def format_alert(
    rule,           # AlertRule ORM row
    scope_key: str, # full scope_key (e.g. "model:claude-opus-4-7")
    value: float,   # the triggering metric value
) -> tuple[str, dict[str, Any]]:
    """Plain-text alert composition. Returns (text, reply_markup_dict)."""
    import hashlib
    scope_hash8 = hashlib.sha256(scope_key.encode()).hexdigest()[:8]
    text = (
        f"🚨 Alert fired: {rule.name}\n\n"
        f"Metric: {rule.metric}\n"
        f"Scope: {scope_key}\n"
        f"Value: {value:.2f}\n"
        f"Threshold: {rule.threshold_fire}"
    )
    kb = _kb([
        [("✓ Ack 1h", f"{CallbackVerb.ack_alert}:{rule.rule_id}:{scope_hash8}")],
    ])
    return text, kb

class AlertAckRequest(BaseModel):
    rule_id: int
    scope_hash: str = Field(..., min_length=8, max_length=8, pattern=r"^[0-9a-f]{8}$")

@router.post("/alerts/_ack", status_code=200)
async def ack_alert(payload: AlertAckRequest, db: AsyncSession = Depends(get_session)):
    """Resolve scope_hash to full scope_key, set state.acked_until = now + 1h."""
    rows = (await db.execute(
        select(AlertState).where(AlertState.rule_id == payload.rule_id)
    )).scalars().all()
    target = None
    for r in rows:
        if hashlib.sha256(r.scope_key.encode()).hexdigest()[:8] == payload.scope_hash:
            target = r
            break
    if target is None:
        raise HTTPException(404, f"no alert_state matching rule_id={payload.rule_id} scope_hash={payload.scope_hash}")
    target.acked_until = datetime.now(UTC).replace(tzinfo=None) + timedelta(hours=1)
    await db.commit()
    return {"ok": True, "acked_until": target.acked_until.isoformat()}
