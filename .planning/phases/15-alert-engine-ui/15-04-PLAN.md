<!-- Standalone fetchers (project convention from Phase 14 — both surfaces): -->

export async function fetchAlertRules(limit = 200, offset = 0): Promise<AlertRuleListResponse> {
  const url = `/api/alerts/rules?limit=${limit}&offset=${offset}`
  return fetchJson<AlertRuleListResponse>(url)
}

export async function fetchAlertRuleCreate(body: AlertRuleCreate): Promise<AlertRule> {
  return fetchJson<AlertRule>('/api/alerts/rules', { method: 'POST', body: JSON.stringify(body) })
}

export async function fetchAlertRulePatch(id: number, body: AlertRulePatch): Promise<AlertRule> {
  return fetchJson<AlertRule>(`/api/alerts/rules/${id}`, { method: 'PATCH', body: JSON.stringify(body) })
}

export async function fetchAlertRuleDelete(id: number): Promise<void> {
  return fetchJson<void>(`/api/alerts/rules/${id}`, { method: 'DELETE' })
}

export async function fetchAlertEvents(range: AlertRange = '7d'): Promise<AlertEventsResponse> {
  return fetchJson<AlertEventsResponse>(`/api/alerts/events?range=${range}`)
}

export async function fetchAlertAck(body: AlertAckRequest): Promise<{ ok: true; acked_until: string }> {
  return fetchJson(`/api/alerts/_ack`, { method: 'POST', body: JSON.stringify(body) })
}

export function useCreateAlertRule() {
  const qc = useQueryClient()
  return useMutation({
    mutationFn: (body: AlertRuleCreate) => api.alertRuleCreate(body),
    onSuccess: () => qc.invalidateQueries({ queryKey: ['alert-rules'] }),
  })
}

export function usePatchAlertRule() {
  const qc = useQueryClient()
  return useMutation({
    mutationFn: ({ id, body }: { id: number; body: AlertRulePatch }) =>
      api.alertRulePatch(id, body),
    // Optimistic ONLY for `enabled` toggle (idempotent transition); pattern
    // mirrors usePatchSchedule (queries.ts:594-623). Non-enabled patches are
    // not optimistic — server may 422 on threshold validation.
    onMutate: async ({ id, body }) => {
      const isEnabledOnly = Object.keys(body).length === 1 && 'enabled' in body
      if (!isEnabledOnly) return undefined
      await qc.cancelQueries({ queryKey: qk.alertRules() })
      const prev = qc.getQueryData<AlertRuleListResponse>(qk.alertRules())
      if (prev) {
        qc.setQueryData<AlertRuleListResponse>(qk.alertRules(), {
          ...prev,
          items: prev.items.map(r => r.rule_id === id ? { ...r, enabled: body.enabled! } : r),
        })
      }
      return { prev }
    },
    onError: (_err, _vars, ctx: any) => {
      if (ctx?.prev) qc.setQueryData(qk.alertRules(), ctx.prev)
    },
    onSettled: () => qc.invalidateQueries({ queryKey: ['alert-rules'] }),
  })
}

export function useDeleteAlertRule() {
  const qc = useQueryClient()
  return useMutation({
    mutationFn: (id: number) => api.alertRuleDelete(id),
    onSuccess: () => qc.invalidateQueries({ queryKey: ['alert-rules'] }),
  })
}

export function useAckAlert() {
  const qc = useQueryClient()
  return useMutation({
    mutationFn: (body: AlertAckRequest) => api.alertAck(body),
    onSuccess: () => qc.invalidateQueries({ queryKey: ['alert-events'] }),
  })
}
