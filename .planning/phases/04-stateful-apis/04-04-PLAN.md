       @router.get("/schedules", response_model=ScheduleListResponse)
       async def list_schedules(
           db: AsyncSession = Depends(get_session),
           limit: int = Query(50, ge=1, le=200),
           offset: int = Query(0, ge=0),
       ):
           q = select(Schedule).order_by(Schedule.created_at.desc()).limit(limit).offset(offset)
           c = select(func.count(Schedule.id))
           rows = (await db.execute(q)).scalars().all()
           total = (await db.execute(c)).scalar_one()
           return ScheduleListResponse(
               items=[ScheduleListItem.model_validate(r) for r in rows],

           )

    2. POST /schedules (SCHD-02) — 201, cron validation + next_run_at compute:
       ```python
       @router.post("/schedules", response_model=ScheduleListItem, status_code=201)
       async def create_schedule(
           payload: ScheduleCreate,
           db: AsyncSession = Depends(get_session),
       ):
           if not validate_cron(payload.cron):
               raise HTTPException(status_code=422, detail="invalid cron expression")
           now = datetime.now(timezone.utc)
           nxt = next_run(payload.cron, now) if payload.enabled else None
           s = Schedule(
               name=payload.name,
               cron=payload.cron.strip(),
               enabled=payload.enabled,
               next_run_at=nxt,
               task_template=payload.task_template,
               skill=payload.skill,
               created_at=now,
               updated_at=now,
           )
           db.add(s)
           try:
               await db.commit()
           except IntegrityError:
               await db.rollback()
               raise HTTPException(status_code=409, detail="schedule name already exists")
           await db.refresh(s)
           return ScheduleListItem.model_validate(s)
       ```

    3. PATCH /schedules/{schedule_id} (SCHD-03) — recompute on cron change OR
       enabled flip (Pitfall 7 + Open Q4: clear AND recompute):
       ```python
       @router.patch("/schedules/{schedule_id}", response_model=ScheduleListItem)
       async def patch_schedule(
           schedule_id: int,
           payload: ScheduleUpdate,
           db: AsyncSession = Depends(get_session),
       ):
           row = (await db.execute(select(Schedule).where(Schedule.id == schedule_id))).scalar_one_or_none()
           if row is None:
               raise HTTPException(status_code=404, detail="schedule not found")
           updates = payload.model_dump(exclude_unset=True)
           if "cron" in updates and updates["cron"] is not None:
               if not validate_cron(updates["cron"]):
                   raise HTTPException(status_code=422, detail="invalid cron expression")
               updates["cron"] = updates["cron"].strip()
           cron_changed = "cron" in updates and updates["cron"] != row.cron
           enabled_changed = "enabled" in updates and updates["enabled"] != row.enabled
           for k, v in updates.items():
               setattr(row, k, v)
           # Recompute / clear next_run_at if either input changed.
           if cron_changed or enabled_changed:
               if row.enabled:
                   row.next_run_at = next_run(row.cron, datetime.now(timezone.utc))
               else:
                   row.next_run_at = None
           row.updated_at = datetime.now(timezone.utc)
           try:
               await db.commit()
           except IntegrityError:
               await db.rollback()
               raise HTTPException(status_code=409, detail="schedule name already exists")
           await db.refresh(row)
           return ScheduleListItem.model_validate(row)
       ```

| Boundary | Description |

| HTTP API ingress | Localhost-only. Pydantic limits cron length (100), description length (500). |
| Anthropic API egress | Outbound to api.anthropic.com only when ANTHROPIC_API_KEY set. AsyncAnthropic constructed inside the call (Pitfall 9), not at module import. |

## STRIDE Threat Register

| Threat ID | Category | Component | Disposition | Mitigation Plan |
|-----------|----------|-----------|-------------|-----------------|
| T-04-04-01 | T (Tampering) | Cron expression injection / DoS via expensive iter | mitigate | croniter is_valid + max_length=100 cap; we only call get_next once per request — bounded work. |
| T-04-04-02 | T (Tampering) | NL prompt injection via SCHD-06 | mitigate | System prompt is locked; model output is validated by croniter.is_valid (RESEARCH Pattern 8 anti-pattern note). Worst case: a valid-but-unintended cron — usability bug, not security. |
