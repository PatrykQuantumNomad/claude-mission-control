```python
import os
from pathlib import Path

import pytest

from cmc.config.settings import Settings


def _write(p: Path, content: str) -> None:
    p.parent.mkdir(parents=True, exist_ok=True)
    p.write_text(content, encoding="utf-8")


def test_settings_anthropic_api_key_default_none():
    """SC4: anthropic_api_key field exists with default None when no env file loads."""
    s = Settings(_env_file=None)
    assert s.anthropic_api_key is None


def test_settings_loads_command_centre_env(tmp_path, monkeypatch):
    """SC3+SC4 root: env_file tuple includes ~/.command-centre/.env."""
    fake_home = tmp_path / "home"
    cmd_env = fake_home / ".command-centre" / ".env"
    _write(cmd_env, "ANTHROPIC_API_KEY=sk-from-cmd\n")
    monkeypatch.setattr(Path, "home", lambda: fake_home)
    # cwd to a directory with NO .env so only ~/.command-centre/.env loads
    monkeypatch.chdir(tmp_path)
    s = Settings()
    assert s.anthropic_api_key == "sk-from-cmd"


def test_settings_env_file_tuple_rightmost_wins(tmp_path, monkeypatch):
    """env_file tuple precedence: ~/.command-centre/.env (rightmost) overrides repo .env."""
    fake_home = tmp_path / "home"
    repo = tmp_path / "repo"
    cmd_env = fake_home / ".command-centre" / ".env"
    repo_env = repo / ".env"
    _write(cmd_env, "ANTHROPIC_API_KEY=sk-from-install\n")
    _write(repo_env, "ANTHROPIC_API_KEY=sk-from-repo\n")
    monkeypatch.setattr(Path, "home", lambda: fake_home)
    monkeypatch.chdir(repo)
    # Also clear any inherited env var that would shadow file loading
    monkeypatch.delenv("ANTHROPIC_API_KEY", raising=False)
    s = Settings()
    assert s.anthropic_api_key == "sk-from-install"
```

## Trust Boundaries

| Boundary | Description |
|----------|-------------|
| `~/.command-centre/.env` → Settings | Install-mode config file becomes a TRUSTED input source for `anthropic_api_key`. Trust is established by file ownership + filesystem permissions; pydantic-settings does NOT verify file mode. |
| Shell env → Settings | UNTRUSTED. `os.environ` may contain stale or malicious values inherited from the shell or parent process. Settings re-reads `.env` regardless, so `os.environ` no longer dictates the trusted value for `anthropic_api_key`. |
| Settings → subprocess (`claude -p`) | Settings-sourced ANTHROPIC_API_KEY is the ONLY trusted source surfaced into the subprocess env. Shell-inherited values are scrubbed first (Pitfall P12). |
| `cmc server` (HTTP) → notifier | Internal trust — notifier consumes its own application's API on localhost. No new trust boundary, just substituting an in-process ORM read for a localhost HTTP read. |
