    ```tsx
    import { ReactNode } from 'react'
    import { UseQueryResult } from '@tanstack/react-query'
    import { Card, CardHeader, CardTitle, CardDescription, CardContent, EmptyState, Skeleton } from './'
    import { ErrorState } from './ErrorState'

    interface PanelCardProps<TData> {
      reqId: string
      title: string
      description?: ReactNode
      trailing?: ReactNode
      query: UseQueryResult<TData, Error>
      empty: { dataNoun: string; when?: (data: TData) => boolean }
      skeleton?: ReactNode
      hiddenWhenEmpty?: boolean
      children: (data: TData) => ReactNode
    }

    function defaultIsEmpty<T>(data: T, when?: (d: T) => boolean): boolean {
      if (when) return when(data)
      const d = data as unknown as { items?: unknown[] }
      return Array.isArray(d?.items) && d.items.length === 0
    }

    function DefaultSkeleton() {
      return (
        <div style={{ display: 'flex', flexDirection: 'column', gap: 'var(--space-sm)' }}>
          <Skeleton variant="text" />
          <Skeleton variant="text" />
          <Skeleton variant="text" />
          <Skeleton variant="text" />
        </div>
      )
    }

    export function PanelCard<T>({ reqId, title, description, trailing, query, empty, skeleton, hiddenWhenEmpty, children }: PanelCardProps<T>) {
      // hiddenWhenEmpty: short-circuit to null (used by AttentionBar)
      if (hiddenWhenEmpty && query.data && defaultIsEmpty(query.data, empty.when)) return null

      return (
        <Card>
          <CardHeader>
            <div className="cmc-panel-card__header">
              <div>
                <CardDescription className="cmc-label">{reqId}</CardDescription>
                <CardTitle>{title}</CardTitle>
                {description && <CardDescription>{description}</CardDescription>}
              </div>
              {trailing && <div>{trailing}</div>}
            </div>
          </CardHeader>
          <CardContent>
            {query.isPending
              ? (skeleton ?? <DefaultSkeleton />)
              : query.isError
              ? <ErrorState message={query.error?.message ?? 'Unknown error'} dataNoun={empty.dataNoun} onRetry={() => query.refetch()} />
              : query.data && defaultIsEmpty(query.data, empty.when)
              ? <EmptyState heading="Nothing to show yet" body={`Once ${empty.dataNoun} arrives it will appear here. Run sync from the header to refresh.`} />
              : query.data
              ? children(query.data)
              : null}
          </CardContent>
        </Card>
      )
    }
    ```

    Step 2 — Create `frontend/src/lib/useFirehose.ts`:
    ```ts
    import { useEffect, useRef, useState } from 'react'

    export interface OtelEvent {
      id: number
      ts: string
      event_name: string
      session_id: string | null
      attrs_mcp_server: string | null
      attrs_mcp_tool: string | null
    }

    export type FirehoseStatus = 'connecting' | 'open' | 'closed'

    interface FirehoseOptions {
      eventName?: string
      since?: string
      bufferSize?: number
      enabled?: boolean
    }

    export function useFirehose(opts: FirehoseOptions = {}) {
      const [events, setEvents] = useState<OtelEvent[]>([])
      const [status, setStatus] = useState<FirehoseStatus>('connecting')
      const lastIdRef = useRef<string | null>(null)

      useEffect(() => {
        if (opts.enabled === false) return
        const params = new URLSearchParams()
        if (opts.eventName) params.set('event_name', opts.eventName)
        if (opts.since)     params.set('since', opts.since)
        const url = `/api/firehose${params.toString() ? `?${params}` : ''}`
        const es = new EventSource(url)
        const cap = opts.bufferSize ?? 500

        const onOpen = () => setStatus('open')
        const onError = () => setStatus('closed')
        const onMessage = (ev: MessageEvent) => {
          lastIdRef.current = ev.lastEventId
          try {
            const evt = JSON.parse(ev.data) as OtelEvent
            setEvents(prev => {
              const next = [...prev, evt]
              return next.length > cap ? next.slice(next.length - cap) : next
            })
          } catch {
            // skip malformed
          }
        }

        es.addEventListener('open', onOpen)
        es.addEventListener('error', onError)
        es.addEventListener('otel', onMessage as EventListener)

        return () => {
          es.removeEventListener('open', onOpen)
          es.removeEventListener('error', onError)
          es.removeEventListener('otel', onMessage as EventListener)
          es.close()
        }
      }, [opts.eventName, opts.since, opts.bufferSize, opts.enabled])

      return { events, status, lastEventId: lastIdRef.current }
    }
    ```
