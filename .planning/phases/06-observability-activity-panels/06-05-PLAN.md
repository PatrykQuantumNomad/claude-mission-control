    Step 1 — `OtelPanel.tsx` (ACTV-03):
    Component shape:
    ```tsx
    import { useState, useMemo } from 'react'
    import { Card, CardHeader, CardTitle, CardDescription, CardContent, StatePill, RelativeTime } from '../ui'
    import { useFirehose } from '../../lib/useFirehose'

    export function OtelPanel() {
      const [filter, setFilter] = useState('')
      const { events, status } = useFirehose({ bufferSize: 500 })

      // Client-side filter (RESEARCH §SSE filter strategy 2)
      const filtered = useMemo(() => {
        if (!filter.trim()) return events
        const f = filter.toLowerCase()
        return events.filter(e => e.event_name.toLowerCase().includes(f))
      }, [events, filter])

      return (
        <Card>
          <CardHeader>
            <div className="cmc-panel-card__header">
              <div>
                <CardDescription className="cmc-label">ACTV-03</CardDescription>
                <CardTitle>OTEL Firehose</CardTitle>
                <CardDescription>{filtered.length}/{events.length} events shown</CardDescription>
              </div>
              <div className="cmc-otel-controls">
                <input type="text" placeholder="filter event_name…" value={filter} onChange={e => setFilter(e.target.value)} className="cmc-input" />
                <StatePill state={status === 'open' ? 'ok' : status === 'connecting' ? 'pending' : 'stale'}>{status}</StatePill>
              </div>
            </div>
          </CardHeader>
          <CardContent>
            <div className="cmc-otel-feed" role="log" aria-live="polite">
              {filtered.length === 0 ? (
                <p className="cmc-text-dim">{status === 'closed' ? 'Reconnecting…' : 'Waiting for events…'}</p>
              ) : (
                // Newest at top (reverse render): better for live monitoring
                [...filtered].reverse().map(e => (
                  <div key={e.id} className="cmc-otel-row">
                    <span className="cmc-mono cmc-text-subtle"><RelativeTime ts={e.ts}/></span>
                    <span className="cmc-mono">{e.event_name}</span>
                    {e.session_id && <span className="cmc-mono cmc-text-dim">{e.session_id.slice(0,8)}…</span>}
                  </div>
                ))
              )}
            </div>
          </CardContent>
        </Card>
      )
    }
    ```
    Note: NOT using PanelCard here because there's no UseQueryResult — the hook is bespoke. The Card composition is hand-rolled to mirror PanelCard's visual contract. Document via inline comment: `// Bespoke shell (not PanelCard) — useFirehose returns a different shape than UseQueryResult.`
