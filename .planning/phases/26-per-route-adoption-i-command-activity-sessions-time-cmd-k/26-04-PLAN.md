```typescript
// RecentRoutesTracker — Phase 26 Plan 04 (SHEL-05, CMDK-04).
//
// Zero-render effect that pushes the current route to cmc.recents.routes
// on every navigation to an IN_SCOPE_ROUTES route.
//
// IN_SCOPE_ROUTES mirrors Phase 25 Plan 10's RecentStateTracker IN_SCOPE_ROUTES
// — the same routes that have validateSearch + saved-view support get
// recents tracking. /sessions/compare uses normalized route id from
// useRouterState's matches API (mirror RecentStateTracker pattern).
//
// Pitfall 8 option (b): accept one-frame flicker. We DO push from useEffect
// (post-render), and RecentlyVisitedSection re-renders on the next
// navigation. The "currently active" route is FILTERED OUT of the display
// list (in RecentlyVisitedSection), so the flicker is invisible —
// the user never sees the route they're standing on in "Recently Visited".

import { useEffect } from 'react'
import { useRouterState } from '@tanstack/react-router'
import { pushRecentRoute } from '../../lib/recents'

const IN_SCOPE_ROUTES = new Set([
  '/',
  '/activity',
  '/sessions/compare',
  '/skills',
  '/cost',
  '/alerts',
])

function normalizeRouteId(pathname: string): string {
  // /skills/foo → /skills/$name (mirror Phase 25's normalizeRouteId helper).
  if (pathname.startsWith('/skills/') && pathname !== '/skills/') {
    return '/skills/$name'
  }
  return pathname
}

export function RecentRoutesTracker() {
  const pathname = useRouterState({ select: (s) => s.location.pathname })
  useEffect(() => {
    const route = normalizeRouteId(pathname)
    // Track dynamic-segment route too (/skills/$name renders as /skills/foo).
    if (!IN_SCOPE_ROUTES.has(route) && route !== '/skills/$name') return
    pushRecentRoute({ route, visitedAt: Date.now() })
  }, [pathname])
  return null
}
```

```typescript
// RecentlyVisitedSection — Phase 26 Plan 04 (SHEL-05).
//
// Sidebar section rendering the top 3 recent routes (NOT 5 — CONTEXT
// narrows to 3 for sidebar; Cmd+K shows 5 — see RESEARCH Open Q #5).
// Filters out the currently-active route (Pitfall 8 option b: cleanest
// UX, sidesteps the one-frame flicker issue).
//
// Section header ALWAYS renders (mirrors PinnedViewsSection pattern from
// Phase 25 Plan 09). Empty body (zero recents) is acceptable — first-time
// users see the header until they navigate around.
//
// Active-state: the routes shown here are by definition NOT the currently-
// active route (filtered out), so no active-state styling needed. Hovering
// each row gets the standard SidebarNavLink hover affordance.

import { useMemo } from 'react'
import {
  Activity,
  Bell,
  DollarSign,
  GitCompare,
  Home,
  Sparkles,
  History,
} from 'lucide-react'
import type { LucideIcon } from 'lucide-react'
import { useRouterState } from '@tanstack/react-router'
import { SidebarSection } from '../shell/SidebarSection'
import { SidebarNavLink } from '../shell/SidebarNavLink'
import { getRecentRoutes } from '../../lib/recents'

const ROUTE_META: Record<string, { label: string; Icon: LucideIcon; to: string }> = {
  '/': { label: 'Home', Icon: Home, to: '/' },
  '/activity': { label: 'Activity', Icon: Activity, to: '/activity' },
  '/sessions/compare': { label: 'Sessions Compare', Icon: GitCompare, to: '/sessions/compare' },
  '/skills': { label: 'Skills', Icon: Sparkles, to: '/skills' },
  '/skills/$name': { label: 'Skill detail', Icon: Sparkles, to: '/skills' /* idx fallback */ },
  '/cost': { label: 'Cost', Icon: DollarSign, to: '/cost' },
  '/alerts': { label: 'Alerts', Icon: Bell, to: '/alerts' },
}

interface Props {
  collapsed?: boolean
}

export function RecentlyVisitedSection({ collapsed = false }: Props) {
  const pathname = useRouterState({ select: (s) => s.location.pathname })
  // Re-read on every render — getRecentRoutes is pure (reads localStorage,
  // no React subscription). useRouterState above provides the re-render
  // trigger on navigation; that's sufficient for v1.
  const top3 = useMemo(() => {
    const all = getRecentRoutes()
    // Filter out the currently-active route (Pitfall 8 option b).
    const normalized = pathname.startsWith('/skills/') && pathname !== '/skills/'
      ? '/skills/$name' : pathname
    return all.filter((r) => r.route !== normalized).slice(0, 3)
  }, [pathname])

  return (
    <SidebarSection title="Recently Visited" data-testid="sidebar-section-recently-visited">
      {top3.map((r) => {
        const meta = ROUTE_META[r.route]
        if (!meta) return null
        return (
          <SidebarNavLink
            key={r.route}
            to={meta.to as any}
            label={meta.label}
            Icon={meta.Icon}
            collapsed={collapsed}
            data-testid={`sidebar-recently-visited-route-${meta.to.replace(/\//g, '-').replace(/^-/, '') || 'home'}`}
          />
        )
      })}
    </SidebarSection>
  )
}
```
