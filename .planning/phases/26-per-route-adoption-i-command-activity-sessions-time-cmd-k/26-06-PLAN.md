For the time-range commands, define a small inline preset list and helpers:

```typescript
// CMDK-03 — Time range presets (mirror TimePicker preset list but condensed).
const CMDK_TIME_PRESETS = [
  { value: '1h', label: 'Last 1 hour', from: 'now-1h', to: 'now' },
  { value: '24h', label: 'Last 24 hours', from: 'now-24h', to: 'now' },
  { value: '7d', label: 'Last 7 days', from: 'now-7d', to: 'now' },
  { value: '30d', label: 'Last 30 days', from: 'now-30d', to: 'now' },
]

function applyTimeRange(from: string, to: string) {
  navigate({
    to: location.pathname as any,
    search: (prev: Record<string, unknown>) => ({ ...prev, time_from: from, time_to: to }),
  })
}

async function onCopyTimeRange() {
  const search = (location.search ?? {}) as Record<string, unknown>
  const timeFrom = typeof search.time_from === 'string' ? search.time_from : undefined
  const timeTo = typeof search.time_to === 'string' ? search.time_to : undefined
  close()
  if (!timeFrom || !timeTo) {
    toast.error('No time range to copy')
    return
  }
  try {
    await navigator.clipboard.writeText(serializeRange(timeFrom, timeTo))
    toast.success('Time range copied')
  } catch {
    toast.error('Could not access clipboard')
  }
}

async function onPasteTimeRange() {
  close()
  try {
    const text = await navigator.clipboard.readText()
    const parsed = parseRangeFromText(text)
    if (!parsed) { toast.error('No time range on clipboard'); return }
    const f = asTimeToken(parsed.time_from)
    const t = asTimeToken(parsed.time_to)
    if (!f || !t) { toast.error('No time range on clipboard'); return }
    applyTimeRange(f, t)
    toast.message('Pasted time range')
  } catch {
    toast.error('Could not access clipboard')
  }
}
```

// CMDK-02 — Density command. Local state for the checkmark display ONLY
// (mirror DensityToggle.tsx pattern). setDensity() writes <html data-density>
// + localStorage + dispatches no event needed (CSS cascade re-paints).
// POLI-11 zero-rerender invariant preserved: setDensity does NOT trigger a
// React subtree re-render below DensityToggle/CommandPalette.
const [currentDensity, setCurrentDensity] = useState<Density>(() =>
  typeof window !== 'undefined' ? getDensity() : 'comfortable',
)

function chooseDensity(d: Density) {
  setDensity(d)
  setCurrentDensity(d)  // local checkmark only
  close()
}

const DENSITIES: { value: Density; label: string }[] = [
