Create `frontend/src/components/time/ChartBrushController.tsx` — a custom hook (not a component) that returns an `onDragEnd` handler bound to the current pathname. Components that want brush-zoom call this hook and pass the resulting handler + their data array to `<Brush />`.

Why a hook not a component: `<Brush />` is a recharts internal that MUST be rendered as a direct child of a cartesian chart (BarChart/LineChart). A wrapper component would break recharts' child-detection. The hook gives panels a generic onDragEnd they can wire directly.

EXACT contents:

```typescript
// ChartBrushController — Phase 26 Plan 05 (TIME-05).
//
// useChartBrush() returns an onDragEnd handler that translates a recharts
// Brush commit ({ startIndex, endIndex }) into a navigate() call writing
// absolute ISO time_from/time_to to the URL. Generic — any time-anchored
// chart can wire it.
//
// The brush commits ABSOLUTE timestamps (per CONTEXT: "zoomed range
// freezes to absolute"). This deliberately triggers AutoRefreshController's
// pause branch (Plan 03) — refresh is meaningless during investigation.
//
// onChange (per-frame during drag) is intentionally NOT consumed — we
// commit on onDragEnd only to avoid a navigate-per-mouse-move spam.

import { useCallback } from 'react'
import { useNavigate, useRouterState } from '@tanstack/react-router'

interface BrushDataRow {
  day: string  // ISO date or 'YYYY-MM-DD' — coerce to full ISO before navigate
}

export interface UseChartBrushOptions<T extends BrushDataRow> {
  data: T[]
}

export function useChartBrush<T extends BrushDataRow>({ data }: UseChartBrushOptions<T>) {
  const navigate = useNavigate()
  const pathname = useRouterState({ select: (s) => s.location.pathname })

  const onDragEnd = useCallback(
    (payload: { startIndex?: number; endIndex?: number } | undefined) => {
      if (!payload) return
      const { startIndex, endIndex } = payload
      if (typeof startIndex !== 'number' || typeof endIndex !== 'number') return
      const lo = Math.min(startIndex, endIndex)
      const hi = Math.max(startIndex, endIndex)
      const fromRow = data[lo]
      const toRow = data[hi]
      if (!fromRow?.day || !toRow?.day) return
      // Coerce 'YYYY-MM-DD' (date-only) to full ISO at start/end-of-day
      // so the URL contains a parseable ISO_ABS token that asTimeToken accepts.
      const fromIso = fromRow.day.includes('T') ? fromRow.day : `${fromRow.day}T00:00:00.000Z`
      const toIso = toRow.day.includes('T') ? toRow.day : `${toRow.day}T23:59:59.999Z`
      navigate({
        to: pathname as any,
        search: (prev: Record<string, unknown>) => ({
          ...prev,
          time_from: fromIso,
          time_to: toIso,
        }),
        replace: false,
      })
    },
    [data, navigate, pathname],
  )

  return { onDragEnd }
}
```

```typescript
// ResetZoomButton — Phase 26 Plan 05 (TIME-05).
//
// Chart-header chrome. Visible only when the URL window is absolute (a
// brush-zoom commit has fired). Click clears time_from/time_to from URL,
// which lets the panel's per-route fallback (Wave 4 plans wire this) take
// over again. Also unfreezes AutoRefreshController.

import { useNavigate, useRouterState } from '@tanstack/react-router'
import { ZoomOut } from 'lucide-react'

const ISO_RE = /^\d{4}-\d{2}-\d{2}T/

export function ResetZoomButton() {
  const navigate = useNavigate()
  const location = useRouterState({ select: (s) => s.location })
  const search = (location.search ?? {}) as Record<string, unknown>
  const timeFrom = typeof search.time_from === 'string' ? search.time_from : undefined
  const isZoomed = ISO_RE.test(timeFrom ?? '')
  if (!isZoomed) return null

  function resetZoom() {
    navigate({
      to: location.pathname as any,
      search: (prev: Record<string, unknown>) => ({
        ...prev,
        time_from: undefined,
        time_to: undefined,
      }),
      replace: false,
    })
  }

  return (
    <button
      type="button"
      className="cmc-reset-zoom-button cmc-btn cmc-btn--ghost"
      data-testid="reset-zoom-button"
      aria-label="Reset zoom"
      onClick={resetZoom}
    >
      <ZoomOut size={14} aria-hidden />
      <span>Reset zoom</span>
    </button>
  )
}
```
