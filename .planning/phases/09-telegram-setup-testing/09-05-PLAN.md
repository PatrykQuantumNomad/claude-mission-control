Theme module contract (~30 LOC):
```ts
// frontend/src/lib/theme.ts
export type Theme = "dark" | "light";
export const DEFAULT_THEME: Theme = "dark";
const KEY = "cmc.theme";

export function getTheme(): Theme {
    if (typeof window === "undefined") return DEFAULT_THEME;
    const v = window.localStorage.getItem(KEY);
    return v === "light" ? "light" : "dark";
}

export function setTheme(theme: Theme): void {
    if (typeof window === "undefined") return;
    window.localStorage.setItem(KEY, theme);
    document.documentElement.dataset.theme = theme;
}

// Apply on initial page load — call from main.tsx OR AppShell mount.
export function applyTheme(): void {
    document.documentElement.dataset.theme = getTheme();
}
```

ThemeToggle component contract (~30 LOC):
```tsx
// frontend/src/components/shell/ThemeToggle.tsx
import { useState, useEffect } from "react";
import { getTheme, setTheme } from "../../lib/theme";

export function ThemeToggle() {
    const [theme, setLocal] = useState<"dark" | "light">("dark");
    useEffect(() => { setLocal(getTheme()); }, []);
    const toggle = () => {
        const next = theme === "dark" ? "light" : "dark";
        setTheme(next);
        setLocal(next);
    };
    return (
        <button
            type="button"
            onClick={toggle}
            data-testid="theme-toggle"
            aria-label={`Switch to ${theme === "dark" ? "light" : "dark"} theme`}
            className="..."
        >
            {theme === "dark" ? <SunIcon /> : <MoonIcon />}
        </button>
    );
}
```
