    Step 3 — Create backend/cmc/telegram/api.py. Five functions, each httpx-based.
    CRITICAL: do NOT accept a `parse_mode` argument anywhere (Pitfall P3 — verified
    by a test in this same file).
    ```python
    """Bot API wrapper. Plain text only — Pitfall P3."""
    from __future__ import annotations
    from typing import Any, Optional

    import httpx

    BASE_URL = "https://api.telegram.org/bot{token}"


    async def get_me(token: str, *, client: Optional[httpx.AsyncClient] = None) -> dict[str, Any]:
        """Verify token. Returns the result dict on success; raises httpx.HTTPStatusError on failure."""
        url = BASE_URL.format(token=token) + "/getMe"
        async with (client or httpx.AsyncClient(timeout=10)) as c:
            r = await c.get(url)
            r.raise_for_status()
            return r.json().get("result", {})


    async def send_message(
        token: str,
        chat_id: str,
        text: str,
        *,
        reply_markup: Optional[dict[str, Any]] = None,
        client: Optional[httpx.AsyncClient] = None,
    ) -> dict[str, Any]:
        """Send plain text. NO parse_mode parameter (Pitfall P3).

        Returns the message dict (incl. message_id) so callers can persist it for edit/strip.
        """
        url = BASE_URL.format(token=token) + "/sendMessage"
        body: dict[str, Any] = {"chat_id": chat_id, "text": text}
        if reply_markup is not None:
            body["reply_markup"] = reply_markup
        async with (client or httpx.AsyncClient(timeout=15)) as c:
            r = await c.post(url, json=body)
            r.raise_for_status()
            return r.json().get("result", {})


    async def answer_callback_query(
        token: str,
        callback_query_id: str,
        text: str = "",
        *,
        client: Optional[httpx.AsyncClient] = None,
    ) -> None:
        """Must be called within 15s of receiving a callback (Telegram contract)."""
        url = BASE_URL.format(token=token) + "/answerCallbackQuery"
        body = {"callback_query_id": callback_query_id, "text": text}
        async with (client or httpx.AsyncClient(timeout=10)) as c:
            r = await c.post(url, json=body)
            r.raise_for_status()


    async def edit_message_reply_markup(
        token: str,
        chat_id: str,
        message_id: str,
        reply_markup: Optional[dict[str, Any]] = None,
        *,
        client: Optional[httpx.AsyncClient] = None,
    ) -> None:
        """Strip inline keyboard after answer recorded so user can't double-press."""
        url = BASE_URL.format(token=token) + "/editMessageReplyMarkup"
        body: dict[str, Any] = {"chat_id": chat_id, "message_id": message_id}
        if reply_markup is None:
            body["reply_markup"] = {}  # empty inline_keyboard — strips buttons
        else:
            body["reply_markup"] = reply_markup
        async with (client or httpx.AsyncClient(timeout=10)) as c:
            r = await c.post(url, json=body)
            r.raise_for_status()


    async def get_updates(
        token: str,
        offset: int,
        timeout: int = 25,
        *,
        client: Optional[httpx.AsyncClient] = None,
    ) -> list[dict[str, Any]]:
        """Long-poll for updates starting from offset. timeout=25 fits inside launchd cycle."""
        url = BASE_URL.format(token=token) + "/getUpdates"
        # client timeout MUST exceed long-poll timeout
        async with (client or httpx.AsyncClient(timeout=timeout + 5)) as c:
            r = await c.get(url, params={"offset": int(offset), "timeout": int(timeout)})
            r.raise_for_status()
            return r.json().get("result", [])
    ```

    """Plain-text Telegram formatters. NO MarkdownV2 (Pitfall P3)."""
    from __future__ import annotations
    from typing import Any, Optional


    def _kb(rows: list[list[tuple[str, str]]]) -> dict[str, Any]:
        """Build inline_keyboard payload from [[(label, callback_data), ...], ...]."""
        return {"inline_keyboard": [[{"text": l, "callback_data": d} for (l, d) in row] for row in rows]}


    def format_decision(decision_row) -> tuple[str, dict[str, Any]]:
        """Decision prompt + Yes/No/Snooze buttons."""
        text = (
            f"❓ Decision needed (#{decision_row.id})\n\n"
            f"{decision_row.prompt}\n\n"
            f"Session: {decision_row.session_id or 'unknown'}"
        )
        kb = _kb([
            [("✅ Yes", f"answer_decision:{decision_row.id}:yes"),
             ("❌ No", f"answer_decision:{decision_row.id}:no")],
            [("⏰ Snooze 30m", f"snooze:decision:{decision_row.id}:30m")],
        ])
        return text, kb


    def format_failure(task_row) -> tuple[str, dict[str, Any]]:
        text = (
            f"💥 Task failed (#{task_row.id})\n\n"
            f"{task_row.title}\n\n"
            f"Error: {(task_row.error_message or 'unknown')[:500]}"
        )
        kb = _kb([
            [("🔄 Rerun", f"rerun_task:{task_row.id}")],
            [("⏰ Snooze 30m", f"snooze:failure:{task_row.id}:30m")],
        ])
        return text, kb


    def format_overdue(schedule_row) -> tuple[str, dict[str, Any]]:
        text = (
            f"⏰ Schedule overdue (#{schedule_row.id})\n\n"
            f"{schedule_row.name}\n"
            f"Cron: {schedule_row.cron}"
        )
        kb = _kb([
            [("⏰ Snooze 30m", f"snooze:overdue_schedule:{schedule_row.id}:30m")],
        ])
        return text, kb


    def format_inbox(inbox_row) -> tuple[str, dict[str, Any]]:

            f"📨 Inbox prompt (#{inbox_row.id})\n\n"
            f"{(inbox_row.prompt or '')[:500]}"
        )
        kb = _kb([
            [("💬 Reply", f"reply_inbox:{inbox_row.id}")],
            [("⏰ Snooze 30m", f"snooze:inbox:{inbox_row.id}:30m")],
        ])

            f"🔐 Approval requested (#{task_row.id})\n\n"

            [("✅ Approve", f"approve_task:{task_row.id}"),

        ])
        return text, kb

    Step 5 — Create backend/cmc/telegram/dash_router.py — pure functions,
    no IO. The `route()` function returns `(METHOD, /api/path, body_dict)`
    that the handler then POSTs/PATCHes via httpx to `http://127.0.0.1:8765`.
    ```python
    """Telegram callback_data → backend HTTP route mapper. Pure function — no IO."""
    from __future__ import annotations
    from typing import Any


    class CallbackParseError(ValueError):
        pass


    def decode_callback(data: str) -> tuple[str, list[str]]:
        """`approve_task:42` -> ('approve_task', ['42']); `snooze:overdue_schedule:7:30m` -> ('snooze', ['overdue_schedule','7','30m'])."""
        if not data:
            raise CallbackParseError("empty callback_data")
        head, *rest = data.split(":")
        return head, rest


    def route(verb: str, args: list[str]) -> tuple[str, str, dict[str, Any]]:
        """Map (verb, args) -> (METHOD, /api/path, body).

        Verbs:
          approve_task:<id>             -> POST /api/tasks/{id}/approve
          reject_task:<id>              -> POST /api/tasks/{id}/reject (or DELETE per Phase 4)
          rerun_task:<id>               -> POST /api/tasks/{id}/rerun
          answer_decision:<id>:<yes|no> -> POST /api/decisions/{id}/answer  body={"answer": "yes|no"}
          reply_inbox:<id>              -> NOOP at routing layer; handler enters reply state
          snooze:<kind>:<entity_id>:<duration> -> PATCH /api/notifications/{notif_id}/snooze?duration=...
              (handler resolves notif_id from notification_log via kind+entity_id+chat_id)
          estop                         -> POST /api/system/emergency-stop body={"reason": "telegram"}
        """
        if verb == "approve_task" and len(args) == 1:
            return ("POST", f"/api/tasks/{args[0]}/approve", {})
        if verb == "reject_task" and len(args) == 1:
            return ("POST", f"/api/tasks/{args[0]}/reject", {})
        if verb == "rerun_task" and len(args) == 1:
            return ("POST", f"/api/tasks/{args[0]}/rerun", {})
        if verb == "answer_decision" and len(args) == 2:
            return ("POST", f"/api/decisions/{args[0]}/answer", {"answer": args[1]})
        if verb == "reply_inbox" and len(args) == 1:
            # handler.py captures the next text message from this user; not a route call
            return ("NOOP", f"/api/inbox/{args[0]}", {})
        if verb == "snooze" and len(args) == 3:
            # handler resolves notif_id from (kind, entity_id, chat_id) before PATCH
            kind, entity_id, duration = args
            return ("RESOLVE_THEN_PATCH", f"/api/notifications/_resolve/{kind}/{entity_id}", {"duration": duration})
        if verb == "estop" and len(args) == 0:
            return ("POST", "/api/system/emergency-stop", {"reason": "telegram"})
        raise CallbackParseError(f"unknown verb {verb!r} with {len(args)} args")
    ```

    """TELE-07: render telegram-notifier.plist OR telegram-handler.plist."""
    from __future__ import annotations
    from importlib.resources import files
    from pathlib import Path
    from string import Template

    _VARIANTS = {"notifier": "com.cmc.telegram-notifier.plist.j2",
                 "handler":  "com.cmc.telegram-handler.plist.j2"}


    def render_plist(variant: str, python_path: str | Path, repo_root_path: str | Path) -> str:
        if variant not in _VARIANTS:
            raise ValueError(f"unknown variant {variant!r}; expected one of {list(_VARIANTS)}")
        py = Path(python_path).resolve()
        rr = Path(repo_root_path).resolve()
        tmpl_text = (files("cmc.telegram.templates") / _VARIANTS[variant]).read_text()
        return Template(tmpl_text).safe_substitute(
            python_path=str(py),
            python_path_dir=str(py.parent),
            repo_root=str(rr),
        )


    def main() -> None:
        """CLI entry: `python -m cmc.telegram.plist_render --variant notifier <python> <root>`."""
        import argparse, sys
        ap = argparse.ArgumentParser()
        ap.add_argument("--variant", choices=list(_VARIANTS.keys()), required=True)
        ap.add_argument("python_path")
        ap.add_argument("repo_root")
        args = ap.parse_args()
        sys.stdout.write(render_plist(args.variant, args.python_path, args.repo_root))


    if __name__ == "__main__":
        main()
    ```

    """TELE-04 backend support: list + snooze notification_log rows."""
    from __future__ import annotations
    from datetime import datetime, timedelta, timezone
    from typing import Optional

    from fastapi import APIRouter, Depends, HTTPException, Query
    from sqlalchemy import select, desc
    from sqlalchemy.ext.asyncio import AsyncSession

    from cmc.db import get_session
    from cmc.db.models.notification_log import NotificationLog

    router = APIRouter(prefix="/notifications", tags=["notifications"])

    _DURATION_TABLE = {"15m": 15, "30m": 30, "1h": 60, "4h": 240}


    @router.get("")
    async def list_notifications(
        kind: Optional[str] = Query(None),
        limit: int = Query(50, ge=1, le=500),
        db: AsyncSession = Depends(get_session),
    ) -> dict:
        q = select(NotificationLog).order_by(desc(NotificationLog.sent_at)).limit(limit)
        if kind:
            q = q.where(NotificationLog.kind == kind)
        rows = (await db.execute(q)).scalars().all()
        return {
            "notifications": [
                {
                    "id": r.id, "kind": r.kind, "entity_id": r.entity_id,
                    "chat_id": r.chat_id, "message_id": r.message_id,
                    "sent_at": r.sent_at.isoformat() if r.sent_at else None,
                    "snoozed_until": r.snoozed_until.isoformat() if r.snoozed_until else None,
                    "status": r.status,
                }
                for r in rows
            ]
        }


    @router.patch("/{notification_id}/snooze")
    async def snooze_notification(
        notification_id: int,
        duration: str = Query("30m"),
        db: AsyncSession = Depends(get_session),
    ) -> dict:
        if duration not in _DURATION_TABLE:
            raise HTTPException(status_code=400, detail=f"duration must be one of {list(_DURATION_TABLE)}")
        row = (await db.execute(
            select(NotificationLog).where(NotificationLog.id == notification_id)
        )).scalar_one_or_none()
        if row is None:
            raise HTTPException(status_code=404, detail="notification not found")
        until = datetime.now(timezone.utc) + timedelta(minutes=_DURATION_TABLE[duration])
        row.snoozed_until = until
        row.status = "snoozed"
        await db.commit()
        return {"id": row.id, "snoozed_until": until.isoformat(), "status": "snoozed"}


    @router.get("/_resolve/{kind}/{entity_id}")
    async def resolve_notification(
        kind: str, entity_id: str,
        chat_id: Optional[str] = Query(None),
        db: AsyncSession = Depends(get_session),
    ) -> dict:
        """Lookup helper for the telegram handler (snooze callback): given
        (kind, entity_id, chat_id) returns the matching notification_log id."""
        q = select(NotificationLog).where(
            (NotificationLog.kind == kind) & (NotificationLog.entity_id == entity_id)
        )
        if chat_id:
            q = q.where(NotificationLog.chat_id == chat_id)
        q = q.order_by(desc(NotificationLog.sent_at)).limit(1)
        row = (await db.execute(q)).scalar_one_or_none()
        if row is None:
            raise HTTPException(status_code=404, detail="no matching notification")
        return {"id": row.id, "kind": row.kind, "entity_id": row.entity_id}
    ```
