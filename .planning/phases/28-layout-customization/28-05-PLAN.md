```
test('split-pane resize persists', async ({ page }) => {
  // Need 2 valid session ids — reuse fixture pattern from existing v13-saved-views.spec.ts session-compare scenario
  const { a, b } = await getCompareSessionIds(page)
  await page.goto(`/sessions/compare?a=${a}&b=${b}`)
  await page.waitForSelector('[data-testid="resize-handle-compare"]')

  const handle = page.locator('[data-testid="resize-handle-compare"]')
  // Move handle 200px right via mouse — Playwright drag
  const box = await handle.boundingBox()
  await page.mouse.move(box.x + box.width / 2, box.y + box.height / 2)
  await page.mouse.down()
  await page.mouse.move(box.x + box.width / 2 + 200, box.y + box.height / 2, { steps: 10 })
  await page.mouse.up()

  // Assert URL gained split_sizes=compare:<a>,<b>
  await expect(page).toHaveURL(/split_sizes=compare:\d+,\d+/)
  const url = new URL(page.url())
  const sizes = url.searchParams.get('split_sizes')
  expect(sizes).toMatch(/^compare:\d+,\d+$/)
  // Verify the sizes are NOT 50,50 (proves the drag took effect)
  const [, pair] = sizes!.split(':')
  const [left] = pair.split(',').map(Number)
  expect(left).toBeGreaterThan(55)  // drag was rightward
})
```
