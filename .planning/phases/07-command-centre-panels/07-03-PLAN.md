    import { createContext, ReactNode, useContext, useEffect, useState, FormEvent } from 'react'
    import { Sheet, Button } from '../ui'
    import { useCreateTask } from '../../lib/queries'
    import { storage } from '../../lib/storage'
    import type { TaskCreate } from '../../lib/api'

    const DRAFT_KEY = 'composer.task.draft'

    interface TaskComposerCtxValue {
      open: boolean
      setOpen: (open: boolean) => void
    }
    const TaskComposerCtx = createContext<TaskComposerCtxValue | null>(null)

    export function TaskComposerProvider({ children }: { children: ReactNode }) {
      const [open, setOpen] = useState(false)
      return (
        <TaskComposerCtx.Provider value={{ open, setOpen }}>
          {children}
          <TaskComposer open={open} onOpenChange={setOpen} />
        </TaskComposerCtx.Provider>
      )
    }

    export function useTaskComposer() {
      const ctx = useContext(TaskComposerCtx)
      if (!ctx) throw new Error('useTaskComposer must be used within TaskComposerProvider')
      return ctx
    }

    interface TaskComposerProps {
      open: boolean
      onOpenChange: (open: boolean) => void
    }
    export function TaskComposer({ open, onOpenChange }: TaskComposerProps) {
      const m = useCreateTask()
      const [form, setForm] = useState<TaskCreate>(() => storage.get<TaskCreate>(DRAFT_KEY) ?? defaultTaskForm())

      // Persist draft on change (Pitfall 6 — distinct key)
      useEffect(() => { storage.set(DRAFT_KEY, form) }, [form])

      // On Sheet open, restore draft (in case it was cleared then re-opened)
      useEffect(() => {
        if (open) {
          const draft = storage.get<TaskCreate>(DRAFT_KEY)
          if (draft) setForm(draft)
        }
      }, [open])

      function handleSubmit(e: FormEvent<HTMLFormElement>) {
        e.preventDefault()
        if (!form.title.trim() || m.isPending) return
        m.mutate(form, {
          onSuccess: () => {
            storage.remove(DRAFT_KEY)
            setForm(defaultTaskForm())
            onOpenChange(false)
          },
        })
      }

      return (
        <Sheet open={open} onOpenChange={onOpenChange} title="New task" description="All fields except title are optional.">
          <form onSubmit={handleSubmit} className="cmc-composer">
            {/* 9 fields per RESEARCH §Pattern 1 + RESEARCH §Backend Endpoint Traceability TPNL-02:
                title (required, maxLength 200), description (textarea),
                model (input), execution_mode (select: interactive | classic | stream — default interactive — RESEARCH §Open Q4 surfaces ALL 3),
                priority (number 1-5, default 3), quadrant (select: do | plan | delegate | drop — nullable),
                risk (select: low | medium | high — nullable), approval (select: auto | awaiting_approval — default auto),
                dry_run (checkbox, default false) */}
            {/* skill (text input, optional) — schedule_id NOT surfaced in v1 (Phase 8 dispatcher concern) */}
            {m.isError ? (
              <p role="alert" className="cmc-composer__error">
                {m.error instanceof Error ? m.error.message : 'Save failed'}
              </p>
            ) : null}
            <div className="cmc-composer__actions">
              <Button type="button" variant="ghost" onClick={() => onOpenChange(false)}>Cancel</Button>
              <Button type="submit" variant="primary" disabled={m.isPending || !form.title.trim()}>
                {m.isPending ? 'Saving…' : 'Create task'}
              </Button>
            </div>
          </form>
        </Sheet>
      )
    }

    function defaultTaskForm(): TaskCreate {
      return {
        title: '', description: '', priority: 3, quadrant: null, approval: 'auto',
        risk: null, dry_run: false, model: null, execution_mode: 'interactive',
        skill: null, scheduled_for: null, schedule_id: null,
      }
    }

    Step 2 — Create `frontend/src/components/panels/TaskBoard.tsx`:
    ```tsx
    import { useState, useMemo } from 'react'
    import { PanelCard, Card, CardHeader, CardTitle, CardContent, Badge, Button, AlertDialog } from '../ui'
    import { useTasks, useApproveTask, useRerunTask, useDeleteTask } from '../../lib/queries'
    import type { TaskListItem } from '../../lib/api'

    export function TaskBoard() {
      const q = useTasks()
      return (
        <PanelCard
          reqId="TPNL-01"
          title="Task Board"
          description="Pending, running, and completed tasks"
          query={q}
          empty={{ dataNoun: 'task data' }}
        >
          {(data) => <TaskBoardBody items={data.items} />}
        </PanelCard>
      )
    }

    function TaskBoardBody({ items }: { items: TaskListItem[] }) {
      const grouped = useMemo(() => groupByStatus(items), [items])
      // grouped = { pending: [], running: [], done: [], awaiting_approval: [], failed: [] }
      // Render layout:
      //   1. Above-board banner if grouped.awaiting_approval.length > 0 (Pitfall 11)
      //   2. 3-column grid: pending | running | done
      //   Failed rows merge into "done" column with a "Failed" StatePill
      return (
        <div className="cmc-task-board">
          {grouped.awaiting_approval.length > 0 && (
            <div className="cmc-task-board__banner" role="status">
              {grouped.awaiting_approval.length} task{grouped.awaiting_approval.length === 1 ? '' : 's'} awaiting your approval
              <ul>{grouped.awaiting_approval.map(t => <ApprovalRow key={t.id} task={t} />)}</ul>
            </div>
          )}
          <div className="cmc-task-board__columns">
            <Column title="Pending" tasks={grouped.pending} />
            <Column title="Running" tasks={grouped.running} />
            <Column title="Done" tasks={[...grouped.done, ...grouped.failed]} />
          </div>
        </div>
      )
    }

    function Column({ title, tasks }: { title: string; tasks: TaskListItem[] }) {
      return (
        <Card>
          <CardHeader><CardTitle>{title} ({tasks.length})</CardTitle></CardHeader>
          <CardContent>
            {tasks.length === 0
              ? <p className="cmc-text-subtle">No tasks</p>
              : <ul>{tasks.map(t => <TaskRow key={t.id} task={t} />)}</ul>}
          </CardContent>
        </Card>
      )
    }

    function TaskRow({ task }: { task: TaskListItem }) {
      const [confirmOpen, setConfirmOpen] = useState(false)
      const approveM = useApproveTask()
      const rerunM = useRerunTask()
      const deleteM = useDeleteTask()
      // Render: title, badges (skill/model/quadrant/risk — omit nulls), action buttons (transition-aware: Rerun only on failed)
      // Delete button opens AlertDialog
      return (
        <li className="cmc-task-board__row">
          <span>{task.title}</span>
          <div className="cmc-task-board__badges">
            {task.skill && <Badge>{task.skill}</Badge>}
            {task.model && <Badge>{task.model}</Badge>}
            {task.quadrant && <Badge>{task.quadrant}</Badge>}
            {task.risk && <Badge variant={task.risk === 'high' ? 'destructive' : 'default'}>{task.risk}</Badge>}
            {task.status === 'failed' && <Badge variant="destructive">Failed</Badge>}
          </div>
          <div className="cmc-task-board__actions">
            {task.status === 'failed' && <Button size="sm" onClick={() => rerunM.mutate(task.id)} disabled={rerunM.isPending}>Rerun</Button>}
            {task.status !== 'running' && <Button size="sm" variant="ghost" onClick={() => setConfirmOpen(true)}>Delete</Button>}
          </div>
          <AlertDialog
            open={confirmOpen}
            onOpenChange={setConfirmOpen}
            title={`Delete task '${task.title}'?`}
            description="This cannot be undone."
            actionLabel="Delete"
            actionVariant="destructive"
            onAction={() => deleteM.mutate(task.id, { onSuccess: () => setConfirmOpen(false) })}
          />
        </li>
      )
    }

    function ApprovalRow({ task }: { task: TaskListItem }) {
      const m = useApproveTask()
      return <li>{task.title} <Button size="sm" onClick={() => m.mutate(task.id)} disabled={m.isPending}>Approve</Button></li>
    }

    function groupByStatus(items: TaskListItem[]) {
      const out: Record<string, TaskListItem[]> = { pending: [], running: [], done: [], awaiting_approval: [], failed: [] }
      for (const t of items) (out[t.status] ?? out.pending).push(t)
      return out
    }
    ```
