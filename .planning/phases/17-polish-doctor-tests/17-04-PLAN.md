    ```typescript
    // TEST-05b — /sessions/compare picker → diff flow.
    //
    // Phase 16 shipped two picker entry points (Plan 16-03):
    //   1. SessionsTable per-row Compare button → /sessions/compare?a={id}
    //   2. Cmd+K context-aware "Compare with…" / "Pick a different session B"
    //      → opens ComparePicker Sheet → click row → /sessions/compare?a=&b=
    //
    // This test exercises BOTH entry points in sequence (button for A, palette
    // for B), then asserts the two-up render lands. Skips when the developer's
    // DB has fewer than 2 sessions — Pitfall 6 from research: e2e tests run
    // against the real local backend (no test-DB seed in playwright.config.ts).

    import { test, expect } from '@playwright/test'

    const API = 'http://127.0.0.1:8765'

    test('TEST-05b: /sessions/compare — pick A via row button + B via Cmd+K', async ({
      page,
      request,
    }) => {
      test.setTimeout(60_000)

      // Pre-flight — confirm ≥2 sessions exist.
      const sRes = await request.get(`${API}/api/sessions?limit=5`)
      expect(sRes.ok()).toBe(true)
      const sBody = await sRes.json()
      const ids: string[] = (sBody.items ?? []).map(
        (s: { session_id: string }) => s.session_id,
      )
      test.skip(
        ids.length < 2,
        `TEST-05b: requires ≥2 sessions in DB, found ${ids.length}. ` +
          `Run \`cmc sync\` to ingest local Claude Code sessions, then re-run.`,
      )
      const [sessionA, sessionB] = ids

      // 1. Visit /sessions and click the row Compare button for session A.
      await page.goto('/sessions')
      // The SessionsTable's per-row Compare button has
      // aria-label="Compare session {session_id}". Use exact match to avoid
      // collisions if multiple rows render.
      const rowCompareA = page.getByRole('button', {
        name: `Compare session ${sessionA}`,
      })
      await expect(rowCompareA).toBeVisible({ timeout: 10_000 })
      await rowCompareA.click()

      // 2. Assert URL contains ?a={sessionA}. The route validateSearch
      //    (Plan 16-02) strips invalid UUIDs, so the value MUST round-trip.
      await expect(page).toHaveURL(new RegExp(`/sessions/compare\\?a=${sessionA}`))

      // 3. Open Cmd+K. Per command-palette.spec.ts, the palette listens on
      //    document and responds to ControlOrMeta+KeyK.
      await page.locator('body').click()
      await page.keyboard.press('ControlOrMeta+KeyK')
      const palette = page.getByRole('dialog', {
        name: 'Mission Control command palette',
      })
      await expect(palette).toBeVisible({ timeout: 5_000 })

      // 4. Click the context-aware Compare action. Per Plan 16-03, the label
      //    is "Compare with…" or "Pick a different session B" depending on
      //    whether ?b is set (we're at ?a only, so it's the former).
      const compareWithItem = palette.getByRole('option', {
        name: /Compare with|Pick a different session B/i,
      })
      await expect(compareWithItem).toBeVisible()
      await compareWithItem.click()

      // 5. The ComparePicker Sheet opens. It lists sessions; we click the
      //    first row that's NOT session A (self-compare guard handles
      //    matching session A by disabling that row — verified Plan 16-04
      //    browser check #5).
      // ComparePicker rows render the session_id in monospace text. Match
      // by text content rather than role since the Sheet doesn't re-use
      // SessionsTable's button affordance. Scope the locator to the open
      // Sheet (role='dialog') so the match doesn't accidentally hit the
      // sessionB id rendered elsewhere on the page (e.g., in a row of the
      // SessionsTable that's still mounted behind the Sheet).
      const pickerRow = page
        .locator('[role="dialog"]')
        .locator(`text=${sessionB}`)
        .first()
      await expect(pickerRow).toBeVisible({ timeout: 5_000 })
      await pickerRow.click()

      // 6. URL should now have both ?a and ?b.
      await expect(page).toHaveURL(
        new RegExp(`/sessions/compare\\?a=${sessionA}.*b=${sessionB}`),
      )

      // 7. Assert two-up render: both KPI strips visible. Plan 16-02's
      //    SessionCompareView renders one KPI strip per side; locate by
      //    role='heading' or by the panel's accessible structure.
      //
      //    The KPI strip is a labeled stat group rendered inside a
      //    PanelCard. We assert by the count of session_id-bearing
      //    elements being ≥2 (one per side) — this is the most robust
      //    selector that doesn't rely on internal class names.
      const sessionAReferences = page.getByText(sessionA, { exact: false })
      const sessionBReferences = page.getByText(sessionB, { exact: false })
      await expect(sessionAReferences.first()).toBeVisible({ timeout: 10_000 })
      await expect(sessionBReferences.first()).toBeVisible({ timeout: 10_000 })

      // 8. Assert there are at least two distinct rendered references —
      //    one per side (URL bar bears both, plus side A header bears A,
      //    side B header bears B → expect ≥2 visible occurrences total).
      const aCount = await sessionAReferences.count()
      const bCount = await sessionBReferences.count()
      expect(aCount).toBeGreaterThanOrEqual(1)
      expect(bCount).toBeGreaterThanOrEqual(1)
    })
    ```

    Implementation notes:
    - DO NOT add new `data-testid` attributes to ComparePicker, SessionCompareView, or KPI strip components. The existing `aria-label` on the SessionsTable row button + role-based palette selector + text-content match for the picker row are sufficient.
    - DO NOT pre-flight `POST /api/sync` to seed sessions. Per researcher Pitfall 6, the developer-resilient choice is `test.skip` with a clear "run cmc sync first" message. CI hardening is deferred (no CI exists for e2e in v1.x).
    - The `test.skip` invocation MUST surface a clear, actionable reason — verifier will grep test output for it.
    - DO NOT click the Sheet's row by aria-label / button — the Sheet's row affordance per Plan 16-03 uses click-on-row, not click-on-button. Scope the text-content match to the open dialog (`page.locator('[role="dialog"]').locator('text=${sessionB}').first()`) so the locator can't accidentally hit the sessionB id rendered elsewhere on the page (e.g., the SessionsTable row still mounted behind the Sheet).
