**Probe 1 — CT-1 coverage (always runs):**
```sql
WITH skill_events AS (
  SELECT
    o.id,
    o.ts,
    o.session_id AS column_session_id,
    o.attrs_skill_name AS column_skill_name,
    (SELECT json_extract(value,'$.value.stringValue')
       FROM json_each(json_extract(o.body,'$.record.attributes'))
      WHERE json_extract(value,'$.key')='skill.name' LIMIT 1) AS body_skill_name,
    (SELECT json_extract(value,'$.value.stringValue')
       FROM json_each(json_extract(o.body,'$.record.attributes'))
      WHERE json_extract(value,'$.key')='session.id' LIMIT 1) AS body_session_id,
    CAST((SELECT json_extract(value,'$.value.stringValue')
            FROM json_each(json_extract(o.body,'$.record.attributes'))
           WHERE json_extract(value,'$.key')='duration_ms' LIMIT 1) AS INTEGER) AS duration_ms
  FROM otel_events o
  WHERE o.event_name = 'skill_activated'
    AND o.ts >= datetime('now', '-30 days')
)
SELECT
  COALESCE(column_skill_name, body_skill_name, '<UNKEYED>') AS skill_name,
  COUNT(*) AS total_events,
  SUM(CASE WHEN duration_ms IS NOT NULL THEN 1 ELSE 0 END) AS with_duration_ms,
  SUM(CASE WHEN column_session_id IS NULL AND body_session_id IS NOT NULL THEN 1 ELSE 0 END) AS pre_phase_13_session_rows,
  SUM(CASE WHEN column_skill_name IS NULL AND body_skill_name IS NOT NULL THEN 1 ELSE 0 END) AS pre_phase_13_skill_rows
FROM skill_events
GROUP BY 1
ORDER BY with_duration_ms DESC;
```

**Probe 2 — CT-3 negative-residual guard (only run if CT-1 passes):**
Use the per-row decomposition SQL from 22-RESEARCH.md §Code Examples "Spike Q-reconciliation probe" (the multi-CTE query starting `WITH skill_events AS (... SELECT skill_id, skill_name, total_ms, total_ms - subagent_ms - tool_ms AS body_ms, subagent_ms, tool_ms, ABS(...) **CT-3 PASS:** every row's `negative_residual == 0`. **CT-3 FAIL:** ≥1 row with `negative_residual = 1`. On FAIL, copy the worst 5 offender rows (lowest body_ms) into SPIKE-FINDINGS.md as counter-examples.

**Probe 3 — CT-4 subagent containment (only run if CT-1 passes):**
For each `Agent` tool row that overlaps a skill_activated window, count parent-session inner tool rows (tool_name NOT IN ('Skill','Agent')) whose started_at falls inside the Agent's own [started_at, started_at+duration_ms) sub-window. Use this query:
```sql
WITH skill_events AS (
  SELECT o.session_id, o.ts AS skill_start,
    CAST((SELECT json_extract(value,'$.value.stringValue')
            FROM json_each(json_extract(o.body,'$.record.attributes'))
           WHERE json_extract(value,'$.key')='duration_ms' LIMIT 1) AS INTEGER) AS total_ms
  FROM otel_events o
  WHERE o.event_name = 'skill_activated'
    AND o.ts >= datetime('now', '-30 days')
),
agents_in_skills AS (
  SELECT t.session_id, t.tool_use_id, t.started_at AS agent_start,
         t.duration_ms AS agent_dur
  FROM skill_events s
  JOIN tools t
    ON t.session_id = s.session_id
   AND t.tool_name = 'Agent'
   AND t.started_at >= s.skill_start
   AND t.started_at <  datetime(s.skill_start, '+' || s.total_ms || ' milliseconds')
   AND t.duration_ms IS NOT NULL
   AND s.total_ms IS NOT NULL
),
inner_tools AS (
  SELECT a.tool_use_id AS agent_tool_use_id,
         COUNT(t2.tool_use_id) AS inner_tool_count
  FROM agents_in_skills a
  LEFT JOIN tools t2
    ON t2.session_id = a.session_id
   AND t2.tool_name NOT IN ('Skill','Agent')
   AND t2.started_at >= a.agent_start
   AND t2.started_at <  datetime(a.agent_start, '+' || a.agent_dur || ' milliseconds')
  GROUP BY a.tool_use_id
)
SELECT
  COUNT(*) AS agent_invocations_in_skills,
  SUM(CASE WHEN inner_tool_count > 0 THEN 1 ELSE 0 END) AS agents_with_inner_tools_in_parent_session,
  MAX(inner_tool_count) AS worst_overlap_count
FROM inner_tools;
```
**CT-4 PASS:** `agents_with_inner_tools_in_parent_session == 0` (subagent inner tools live in subagent's own session, as assumed). **CT-4 FAIL:** ≥1 — double-count risk is real, formula must be re-derived (descope, do NOT improvise).
