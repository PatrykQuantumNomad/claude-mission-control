      ```python
      """Cross-cutting dispatcher state: PID files, tick stamp, MAX_CONCURRENT.

      All write paths are atomic (os.replace for PID files; UPSERT for tick stamp)
      to avoid Pitfall 10 (mid-spawn SIGTERM losing the PID reference).
      """
      from __future__ import annotations
      import os
      from datetime import datetime, timezone
      from pathlib import Path
      from typing import Optional

      import psutil
      from sqlalchemy.dialects.sqlite import insert as sqlite_insert

      from cmc.config import load_settings
      from cmc.core.process import pid_dir as _phase4_pid_dir
      from cmc.db.models.system_state import SystemState

      def pid_dir() -> Path:
          """Re-export Phase 4 pid_dir; centralizes the import for dispatcher modules."""
          return _phase4_pid_dir()

      def write_pid_file(task_id: int, pid: int) -> Path:
          """Atomically write {pid_dir}/{task_id}.pid via tmp + os.replace.

          Pitfall 10: must happen IMMEDIATELY after Popen, before any wait/read.
          """
          d = pid_dir()
          d.mkdir(parents=True, exist_ok=True)
          final = d / f"{int(task_id)}.pid"
          tmp = d / f".{int(task_id)}.pid.tmp"
          tmp.write_text(str(int(pid)))
          os.replace(tmp, final)
          return final

      def unlink_pid_file(task_id: int) -> None:
          try:
              (pid_dir() / f"{int(task_id)}.pid").unlink()
          except FileNotFoundError:
              pass

      def list_live_pids() -> set[int]:
          """Set of PIDs from .pid files that psutil considers alive."""
          d = pid_dir()
          if not d.exists():
              return set()
          live: set[int] = set()
          for f in d.glob("*.pid"):
              try:
                  pid = int(f.read_text().strip())
              except (ValueError, FileNotFoundError):
                  continue
              if psutil.pid_exists(pid):
                  live.add(pid)
          return live

      async def stamp_tick(sessions) -> None:
          """UPSERT system_state.dispatcher_last_tick_at = now isoformat. Pitfall 5: caller wraps in try/finally."""
          now = datetime.now(timezone.utc)
          async with sessions() as db:
              await db.execute(
                  sqlite_insert(SystemState)
                  .values(key="dispatcher_last_tick_at", value=now.isoformat(), updated_at=now)
                  .on_conflict_do_update(
                      index_elements=["key"],
                      set_={"value": now.isoformat(), "updated_at": now},
                  )
              )
              await db.commit()

      def max_concurrent() -> int:
          return load_settings().dispatcher_max_concurrent

      MAX_CONCURRENT = 3  # historical default; runtime callers should prefer max_concurrent()
      ```

    Step 1 — Create backend/cmc/dispatcher/sweep.py:
      ```python
      """DISP-03: prune stale .pid files; return live-PID set for concurrency cap input."""
      from __future__ import annotations
      import logging

      import psutil

      from cmc.dispatcher.state import pid_dir

      log = logging.getLogger(__name__)

      def sweep_stale_pids() -> set[int]:
          d = pid_dir()
          if not d.exists():
              return set()
          live: set[int] = set()
          for f in sorted(d.glob("*.pid")):
              try:
                  pid = int(f.read_text().strip())
              except (ValueError, FileNotFoundError):
                  # Garbage or race-deleted; clean up
                  try:
                      f.unlink()
                  except FileNotFoundError:
                      pass
                  continue
              if psutil.pid_exists(pid):
                  live.add(pid)
              else:
                  try:
                      f.unlink()
                  except FileNotFoundError:
                      pass
                  log.info("dispatcher.sweep.unlinked_stale", extra={"pid": pid, "file": str(f)})
      ```

      ```python
      """DISP-01 atomic claim under SQLite WAL. BEGIN IMMEDIATE serializes overlapping cycles."""
      from __future__ import annotations
      from datetime import datetime, timezone

      from sqlalchemy import text
      from sqlalchemy.ext.asyncio import AsyncSession

      _CLAIM_SQL = text("""
          UPDATE tasks
          SET status = 'running',
              started_at = :now
          WHERE id IN (
              SELECT id FROM tasks
              WHERE status = 'pending'
                AND (scheduled_for IS NULL OR scheduled_for <= :now)
              ORDER BY priority ASC, created_at ASC
              LIMIT :slots
          )
          RETURNING *
      """)

      async def claim_pending_tasks(db: AsyncSession, slots: int) -> list[dict]:
          if slots <= 0:
              return []
          now = datetime.now(timezone.utc)
          # Default async session opens a transaction lazily; force IMMEDIATE so the
          # write lock is acquired during the read phase too — without this, two
          # overlapping cycles could each SELECT the same 3 rows before either UPDATEs.
          await db.execute(text("BEGIN IMMEDIATE"))
          try:
              result = await db.execute(_CLAIM_SQL, {"now": now, "slots": int(slots)})
              rows = [dict(r) for r in result.mappings().all()]
              await db.commit()
              return rows
          except Exception:
              await db.rollback()
              raise
      ```
      NOTE: SQLAlchemy 2.0 async session.execute auto-begins a deferred transaction. Calling `BEGIN IMMEDIATE` after that may error with "already in transaction" — if so, switch to acquiring the connection directly: `async with db.bind.connect() as conn: await conn.execute(text("BEGIN IMMEDIATE")); ...`. Verify in the test fixture and adjust. The contract (atomic claim with no double-claim) is what matters; the SQL-emission mechanics may need this fallback.

      ```python
      """DISP-01 schedules→tasks materializer. Pitfall 7: tolerate bad task_template per row."""
      from __future__ import annotations
      import logging
      from datetime import datetime, timezone

      from sqlalchemy import select
      from sqlalchemy.ext.asyncio import AsyncSession

      from cmc.db.models.schedules import Schedule
      from cmc.db.models.tasks import Task
      from cmc.schedules.cron import next_run

      log = logging.getLogger(__name__)

      async def materialize_due_schedules(db: AsyncSession) -> list[int]:
          now = datetime.now(timezone.utc)
          rows = (
              await db.execute(
                  select(Schedule)
                  .where(Schedule.enabled == True)  # noqa: E712 — explicit for SQLite
                  .where(Schedule.next_run_at <= now)
              )
          ).scalars().all()

          created: list[int] = []
          for s in rows:
              try:
                  t = Task(
                      **(s.task_template or {}),
                      schedule_id=s.id,
                      status="pending",
                      created_at=now,
                      scheduled_for=now,
                  )
              except TypeError as exc:
                  # Pitfall 7: bad task_template (typo, unknown column).
                  # Do NOT advance next_run_at — leaves the lag visible via SAPI-04.
                  log.error("dispatcher.materialize.bad_template", extra={"schedule_id": s.id, "err": str(exc)})
                  continue
              db.add(t)
              await db.flush()
              created.append(t.id)
              s.last_run_at = now
              try:
                  s.next_run_at = next_run(s.cron, now)
              except Exception as exc:
                  log.error("dispatcher.materialize.bad_cron", extra={"schedule_id": s.id, "err": str(exc)})
                  # Defensive: leave next_run_at where it was so we don't loop on this schedule.
                  s.next_run_at = None

          await db.commit()
          return created
      ```

      ```python
      """DISP-01 one-cycle orchestrator. Plan 04 finalizes per-task fan-out."""

      import logging

      from sqlalchemy import select

      from cmc.config import load_settings
      from cmc.db import create_engine_for_settings, make_sessionmaker
      from cmc.db.models.system_state import SystemState
      from cmc.dispatcher.claim import claim_pending_tasks
      from cmc.dispatcher.materialize import materialize_due_schedules
      from cmc.dispatcher.state import stamp_tick
      from cmc.dispatcher.sweep import sweep_stale_pids

      log = logging.getLogger(__name__)

      MAX_CONCURRENT = 3  # historical constant; runtime resolves via Settings


      async def run_one_cycle() -> int:
          """One launchd-driven heartbeat. Returns 0 on success, nonzero on cycle-level failure.

          Order:
            1. Build engine + sessionmaker (per-cycle; FastAPI's engine lives elsewhere).
            2. Stamp tick stamp FIRST (Pitfall 5 — try/finally so SAPI-04 sees liveness on partial failure).
            3. Emergency stop check → early return 0 if flag is '1'.
            4. Sweep stale PIDs → set[int] of live pids.
            5. Materialize due schedules (writes new pending Task rows).
            6. Claim up to (max_concurrent - len(live_pids)) pending tasks atomically.
            7. Per-task fan-out — TODO Plan 04 (currently logs the count and returns).
          """
          settings = load_settings()
          engine = create_engine_for_settings(settings)
          sessions = make_sessionmaker(engine)
          try:
              try:
                  await stamp_tick(sessions)
                  async with sessions() as db:
                      row = (await db.execute(
                          select(SystemState).where(SystemState.key == "emergency_stop")
                      )).scalar_one_or_none()
                  if row is not None and (row.value or "") == "1":
                      log.info("dispatcher.emergency_stop_active")
                      return 0

                  live_pids = sweep_stale_pids()

                  async with sessions() as db:
                      new_task_ids = await materialize_due_schedules(db)
                  if new_task_ids:
                      log.info("dispatcher.materialized", extra={"count": len(new_task_ids)})

                  slots = max(0, settings.dispatcher_max_concurrent - len(live_pids))
                  if slots == 0:
                      log.info("dispatcher.no_slots", extra={"live": len(live_pids)})
                      return 0

                  async with sessions() as db:
                      claimed = await claim_pending_tasks(db, slots)
                  log.info("dispatcher.claimed", extra={"count": len(claimed)})

                  # TODO(Plan 04): per-task fan-out. For Plan 01 we end here — the
                  # claimed rows stay in 'running' state until Plan 04 wires
                  # run_classic / run_stream + finalize transition.
                  return 0
              finally:
                  # Plan 01 contract: the tick stamp at the TOP runs even on exception.
                  # Don't add an additional stamp here — duplicate UPSERT is fine but noisy.
                  pass
          finally:
              await engine.dispose()
      ```
