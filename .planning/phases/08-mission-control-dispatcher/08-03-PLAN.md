      ```python
      """DISP-07 marker parser. PITFALLS.md §6: fenced-code-aware, line-start anchored.

      Grammar (LOCKED by Phase 8 RESEARCH §6.7):
        ^\\s*(DECISION|INBOX):\\s+(.*\\S)\\s*$  on assistant text outside fenced code blocks.
      """
      from __future__ import annotations
      import re
      from dataclasses import dataclass
      from typing import Iterator, Literal

      _DECISION_RE = re.compile(r"^\s*DECISION:\s*(?P<body>.*\S)\s*$")
      _INBOX_RE    = re.compile(r"^\s*INBOX:\s*(?P<body>.*\S)\s*$")
      _FENCE_RE    = re.compile(r"^\s*```")


      @dataclass
      class Marker:
          kind: Literal["DECISION", "INBOX"]
          body: str  # body text after marker prefix, stripped


      class MarkerParser:
          """Stateful — call feed_text per chunk, flush at end-of-stream.

          Tracks `in_fence` boolean across deltas; suppresses markers inside fenced
          code blocks. Tracks `_buffer` so a marker spanning a chunk boundary is
          emitted correctly (e.g. 'DECISI' in chunk N + 'ON: body' in chunk N+1).
          """
          def __init__(self) -> None:
              self.in_fence = False
              self._buffer = ""

          def feed_text(self, text: str) -> Iterator[Marker]:
              self._buffer += text
              while "\n" in self._buffer:
                  line, self._buffer = self._buffer.split("\n", 1)
                  yield from self._parse_line(line)

          def flush(self) -> Iterator[Marker]:
              if self._buffer:
                  yield from self._parse_line(self._buffer)
                  self._buffer = ""

          def _parse_line(self, line: str) -> Iterator[Marker]:
              if _FENCE_RE.match(line):
                  self.in_fence = not self.in_fence
                  return
              if self.in_fence:
                  return
              m = _DECISION_RE.match(line)
              if m:
                  yield Marker(kind="DECISION", body=m.group("body").strip())
                  return
              m = _INBOX_RE.match(line)
              if m:
                  yield Marker(kind="INBOX", body=m.group("body").strip())
      ```

      """DISP-07 decision answer poll. Returns answer string or None on timeout."""
      from __future__ import annotations
      import asyncio
      import logging
      from datetime import datetime, timezone
      from typing import Optional

      from sqlalchemy import select

      from cmc.db.models.decisions import Decision

      log = logging.getLogger(__name__)


      async def wait_for_answer(
          sessions,
          decision_id: int,
          *,
          timeout_s: float = 3600,
          poll_s: float = 2.0,
      ) -> Optional[str]:
          """Poll decisions table every poll_s seconds; returns answer when status='answered'.

          Pitfall: open a fresh session per iteration so the connection isn't pinned
          across the sleep (could pile up sleeping connections under load).
          """
          deadline = datetime.now(timezone.utc).timestamp() + timeout_s
          while datetime.now(timezone.utc).timestamp() < deadline:
              async with sessions() as db:
                  row = (
                      await db.execute(select(Decision).where(Decision.id == decision_id))
                  ).scalar_one_or_none()
                  if row is not None and row.status == "answered":
                      return row.answer  # may be None if user explicitly skipped
              await asyncio.sleep(poll_s)
          log.info("dispatcher.answer_poll.timeout", extra={"decision_id": decision_id})
          return None
      ```

      do not bypass via direct DB INSERT — preserves Pydantic validation + future side effects)."""
      from __future__ import annotations
      import logging
      from typing import Optional

      import httpx

      log = logging.getLogger(__name__)


      async def post_inbox_marker(body: str, *, port: int, host: str = "127.0.0.1") -> Optional[int]:
          """POST /api/inbox with {source: 'agent_marker', body}. Returns inbox id on 201, None on error.

          Localhost-only call; tolerates connection errors so a dead server (e.g. during
          dev restart) doesn't kill the stream loop.
          """
          url = f"http://{host}:{int(port)}/api/inbox"
          payload = {"source": "agent_marker", "body": body}
          try:
              async with httpx.AsyncClient(timeout=5.0) as client:
                  resp = await client.post(url, json=payload)
              if resp.status_code in (200, 201):
                  data = resp.json()
                  return int(data.get("id")) if isinstance(data, dict) and "id" in data else None
              log.warning("dispatcher.inbox_post.unexpected_status", extra={"status": resp.status_code, "url": url})
              return None
          except httpx.RequestError as exc:
              log.warning("dispatcher.inbox_post.connection_error", extra={"url": url, "err": str(exc)})
              return None
      ```

      """Stream-mode fake claude. Emits NDJSON, optionally reads NDJSON from stdin.

      Flags:
        --emit-decision BODY        Emit DECISION: BODY then PAUSE (read stdin until EOF or {type:user})
        --emit-inbox BODY           Emit INBOX: BODY (no pause)
        --emit-fenced-decision      Emit a DECISION inside a ```fenced``` block (Pitfall 4 test)
        --emit-multi-marker         Emit DECISION + INBOX in sequence
        --exit-code N               Exit with this code at end-of-stream
        --print-pid-file PATH       Write os.getpid() to PATH on first emit
        --hang                      Sleep 30s
      """
      from __future__ import annotations
      import argparse
      import json
      import os
      import sys
      import time
      from typing import Any


      def emit(event: dict[str, Any]) -> None:
          sys.stdout.write(json.dumps(event) + "\n")
          sys.stdout.flush()


      def assistant_text(text: str) -> dict[str, Any]:
          return {
              "type": "assistant",
              "message": {
                  "role": "assistant",
                  "model": "fake-claude",
                  "content": [{"type": "text", "text": text}],
              },
              "session_id": "fake-session",
              "uuid": "fake-uuid",
          }


      def main() -> int:
          parser = argparse.ArgumentParser(add_help=False)
          parser.add_argument("-p", dest="prompt", default="")
          parser.add_argument("--output-format", default="stream-json")
          parser.add_argument("--input-format", default="stream-json")
          parser.add_argument("--verbose", action="store_true")
          parser.add_argument("--include-partial-messages", action="store_true")
          parser.add_argument("--bare", action="store_true")
          parser.add_argument("--model", default="default")
          parser.add_argument("--emit-decision", default=None)
          parser.add_argument("--emit-inbox", default=None)
          parser.add_argument("--emit-fenced-decision", action="store_true")
          parser.add_argument("--emit-multi-marker", action="store_true")
          parser.add_argument("--exit-code", type=int, default=0)
          parser.add_argument("--print-pid-file")
          parser.add_argument("--hang", action="store_true")
          args, _unknown = parser.parse_known_args()

          if args.print_pid_file:
              with open(args.print_pid_file, "w") as f:
                  f.write(str(os.getpid()))

          # init
          emit({"type": "system", "subtype": "init"})

          if args.hang:
              time.sleep(30)
              return 0

          if args.emit_fenced_decision:
              emit(assistant_text("Some prose\n```\nDECISION: ignored?\n```\nNot inside fence: DECISION: real?"))
              # Allow the parser to flush and the parent to write a follow-up if it wants.
              # Block on stdin briefly (at most 2s) to simulate decision pause.
              try:
                  sys.stdin.read(0)
              except Exception:
                  pass

          if args.emit_decision:
              emit(assistant_text(f"DECISION: {args.emit_decision}"))
              # Pause until stdin EOF or one user message arrives.
              try:
                  for line in sys.stdin:
                      data = json.loads(line)
                      if data.get("type") == "user":
                          break
              except (json.JSONDecodeError, KeyboardInterrupt, BrokenPipeError):
                  pass

          if args.emit_inbox:
              emit(assistant_text(f"INBOX: {args.emit_inbox}"))

          if args.emit_multi_marker:
              emit(assistant_text("DECISION: should I do A?\nINBOX: heads up about B"))

          emit({"type": "result", "subtype": "complete"})
          return args.exit_code


      if __name__ == "__main__":
          raise SystemExit(main())
      ```

      """DISP-06/07/08 stream-mode runner. Pipes both directions; reader thread parses NDJSON.

      The reader runs in a thread (NOT asyncio) so it can block on proc.stdout.readline.
      It dispatches DECISION/INBOX markers to async helpers via asyncio.run_coroutine_threadsafe
      against a per-task event loop spawned in this thread.

      Plan 04 will layer follow-up file→stdin injection on top of this. For Plan 03,
      stdin is opened (PIPE) but only used for decision-timeout 'interrupt' messages.
      """
      from __future__ import annotations
      import asyncio
      import hashlib
      import json
      import logging
      import os
      import subprocess
      import threading
      from datetime import datetime, timezone
      from pathlib import Path
      from typing import Any, Mapping, Optional

      from sqlalchemy import update as _upd

      from cmc.core.paths import repo_root
      from cmc.db.models.decisions import Decision
      from cmc.db.models.tasks import Task
      from cmc.dispatcher.answer_poll import wait_for_answer
      from cmc.dispatcher.inbox_post import post_inbox_marker
      from cmc.dispatcher.marker_parser import Marker, MarkerParser
      from cmc.dispatcher.model_resolve import resolve_model
      from cmc.dispatcher.state import unlink_pid_file, write_pid_file

      log = logging.getLogger(__name__)


      def run_stream(
          task_row: Mapping[str, Any],
          settings,
          sessions,
          *,
          skill: Optional[Any] = None,
      ) -> None:
          """One stream-mode subprocess. Lives in a thread spawned by Plan 04 fan-out."""
          task_id = int(task_row["id"])
          model = resolve_model(task_row, skill, settings)

          log_dir = repo_root() / ".tmp" / "mission-control-queue" / "dispatcher-logs"
          log_dir.mkdir(parents=True, exist_ok=True)
          log_path = log_dir / f"task-{task_id}-stream-{int(datetime.now().timestamp())}.log"

          env = os.environ.copy()
          env.pop("ANTHROPIC_API_KEY", None)  # Pitfall 8

          prompt = task_row.get("description") or task_row.get("title") or ""
          cmd = [
              str(settings.claude_bin),
              "-p", prompt,
              "--output-format", "stream-json",
              "--input-format", "stream-json",
              "--verbose",
              "--include-partial-messages",
              "--model", model,
          ]

          proc = None
          log_fp = open(log_path, "ab", buffering=0)
          # State shared between this function and helpers below.
          loop = asyncio.new_event_loop()
          stop_event = threading.Event()
          decision_timed_out = {"v": False}

          def handle_marker(m: Marker) -> None:
              """Called from the reader thread for each emitted marker."""
              if m.kind == "DECISION":
                  did = _insert_decision(task_id, m.body, sessions, loop)
                  if did is None:
                      log.warning("dispatcher.run_stream.decision_insert_failed", extra={"task_id": task_id})
                      return
                  ans = asyncio.run_coroutine_threadsafe(
                      wait_for_answer(
                          sessions, did,
                          timeout_s=settings.dispatcher_decision_timeout_s,
                          poll_s=settings.dispatcher_answer_poll_s,
                      ),
                      loop,
                  ).result()
                  if ans is None:
                      decision_timed_out["v"] = True
                      # Send 'interrupt' user message into stdin so claude exits its read loop.
                      try:
                          if proc and proc.stdin and not proc.stdin.closed:
                              interrupt_msg = json.dumps({
                                  "type": "user",
                                  "message": {"role": "user", "content": "[dispatcher: decision timeout — aborting]"},
                              }) + "\n"
                              proc.stdin.write(interrupt_msg)
                              proc.stdin.flush()
                      except (BrokenPipeError, ValueError):
                          pass
                      stop_event.set()
              else:  # INBOX
                  asyncio.run_coroutine_threadsafe(
                      post_inbox_marker(m.body, port=settings.port),
                      loop,
                  ).result()

          def _reader() -> None:
              parser = MarkerParser()
              try:
                  assert proc is not None and proc.stdout is not None
                  for line in iter(proc.stdout.readline, ""):
                      if not line:
                          break
                      try:
                          event = json.loads(line)
                      except json.JSONDecodeError:
                          continue
                      etype = event.get("type")
                      if etype == "assistant":
                          for block in event.get("message", {}).get("content", []) or []:
                              if block.get("type") == "text":
                                  for marker in parser.feed_text(block.get("text", "")):
                                      handle_marker(marker)
                      elif etype == "stream_event":
                          ev = event.get("event", {}) or {}
                          if ev.get("type") == "content_block_delta":
                              delta = ev.get("delta", {}) or {}
                              if delta.get("type") == "text_delta":
                                  for marker in parser.feed_text(delta.get("text", "")):
                                      handle_marker(marker)
                      # 'system', 'user', 'result' — ignore for marker purposes
                  for marker in parser.flush():
                      handle_marker(marker)
              except Exception:
                  log.exception("dispatcher.run_stream.reader_error")

          # Run the asyncio loop in a dedicated thread so handle_marker can submit
          # coroutines via run_coroutine_threadsafe.
          loop_thread = threading.Thread(target=_run_loop_forever, args=(loop,), daemon=True)
          loop_thread.start()

          try:
              try:
                  proc = subprocess.Popen(
                      cmd,
                      cwd=str(repo_root()),
                      stdin=subprocess.PIPE,
                      stdout=subprocess.PIPE,
                      stderr=log_fp,
                      env=env,
                      close_fds=True,
                      bufsize=1,        # line-buffered (Pitfall 11)
                      text=True,        # str instead of bytes
              )
              except FileNotFoundError as exc:
                  log.error("dispatcher.run_stream.spawn_failed", extra={"task_id": task_id, "err": str(exc)})
                  _mark_status_sync(task_id, "failed", f"spawn failed: {exc}", sessions, loop)
                  return

              # Pitfall 10: PID file IMMEDIATELY.
              write_pid_file(task_id, proc.pid)

              reader = threading.Thread(target=_reader, daemon=False)
              reader.start()

              proc.wait()
              reader.join(timeout=10)

              if decision_timed_out["v"]:
                  _mark_status_sync(task_id, "failed", "decision timeout", sessions, loop)
              elif proc.returncode != 0:
                  _mark_status_sync(task_id, "failed", f"nonzero exit {proc.returncode}", sessions, loop)
              else:
                  _mark_status_sync(task_id, "done", None, sessions, loop)
          finally:
              try:
                  if proc and proc.stdin and not proc.stdin.closed:
                      proc.stdin.close()
              except Exception:
                  pass
              try:
                  log_fp.close()
              except Exception:
                  pass
              unlink_pid_file(task_id)
              # Stop the loop and join its thread.
              try:
                  loop.call_soon_threadsafe(loop.stop)
              except Exception:
                  pass
              loop_thread.join(timeout=2)
              try:
                  loop.close()
              except Exception:
                  pass


      def _run_loop_forever(loop: asyncio.AbstractEventLoop) -> None:
          asyncio.set_event_loop(loop)
          loop.run_forever()


      def _insert_decision(task_id: int, body: str, sessions, loop: asyncio.AbstractEventLoop) -> Optional[int]:
          """Synchronously insert a Decision row from the reader thread; return new id."""
          dedup = hashlib.sha256(f"{task_id}:{body}".encode()).hexdigest()[:32]
          async def _insert() -> Optional[int]:
              now = datetime.now(timezone.utc)
              async with sessions() as db:
                  d = Decision(
                      task_id=task_id,
                      dedup_key=dedup,
                      content=body,
                      status="pending",
                      created_at=now,
                  )
                  db.add(d)
                  try:
                      await db.commit()
                      return d.id
                  except Exception:
                      await db.rollback()
                      # Probably HITL-02 partial-unique conflict — fetch existing pending row.
                      from sqlalchemy import select
                      async with sessions() as db2:
                          row = (await db2.execute(
                              select(Decision).where(Decision.dedup_key == dedup, Decision.status == "pending")
                          )).scalar_one_or_none()
                          return row.id if row else None
          return asyncio.run_coroutine_threadsafe(_insert(), loop).result()


      def _mark_status_sync(task_id: int, status: str, error_message: Optional[str], sessions, loop: asyncio.AbstractEventLoop) -> None:
          async def _mark() -> None:
              now = datetime.now(timezone.utc)
              values: dict[str, Any] = {"status": status, "ended_at": now}
              if error_message is not None:
                  values["error_message"] = error_message
              async with sessions() as db:
                  await db.execute(_upd(Task).where(Task.id == task_id).values(**values))
                  await db.commit()
          asyncio.run_coroutine_threadsafe(_mark(), loop).result()
      ```

      """Wave-2 spike for RESEARCH §A2 / §Open Q2.

      Verifies claude CLI accepts the symmetric NDJSON shape on stdin so DISP-09's
      follow-up pump (Plan 04) can rely on it. If this returns ('rejected', ...) we
      fall back to RESEARCH §A2 contingency (a) introduce claude-agent-sdk dep, (b)
      use --resume per follow-up, or (c) document and defer DISP-09.

      Run manually: `python -m cmc.dispatcher._input_format_spike` — exits 0 on accept,
      1 on reject, 77 (EX_NOPERM-ish) on skip.
      """
      from __future__ import annotations
      import argparse
      import json
      import os
      import shutil
      import subprocess
      import sys
      import time
      from typing import Literal


      Outcome = Literal["accepted", "rejected", "skipped"]


      def probe_stdin_shape(claude_bin: str | None = None, *, deadline_s: float = 5.0) -> tuple[Outcome, str]:
          binpath = claude_bin or shutil.which("claude")
          if not binpath or not os.path.exists(binpath):
              return ("skipped", "claude binary not present")
          msg = json.dumps({
              "type": "user",
              "message": {"role": "user", "content": "hello from spike"},
          }) + "\n"
          proc = subprocess.Popen(
              [binpath, "-p", "", "--output-format", "stream-json", "--input-format", "stream-json", "--verbose"],
              stdin=subprocess.PIPE,
              stdout=subprocess.PIPE,
              stderr=subprocess.PIPE,
              bufsize=1,
              text=True,
          )
          try:
              proc.stdin.write(msg)
              proc.stdin.flush()
              proc.stdin.close()
              start = time.time()
              while time.time() - start < deadline_s:
                  line = proc.stdout.readline()
                  if not line:
                      break
                  try:
                      event = json.loads(line)
                  except json.JSONDecodeError:
                      continue
                  if event.get("type") == "assistant":
                      proc.terminate()
                      return ("accepted", "assistant event observed")
                  if event.get("type") == "result" and event.get("subtype") == "error":
                      proc.terminate()
                      return ("rejected", f"result error: {event}")
              proc.terminate()
              return ("rejected", "no assistant event within deadline")
          finally:
              try:
                  proc.kill()
              except ProcessLookupError:
                  pass


      def main() -> int:
          outcome, detail = probe_stdin_shape()
          sys.stdout.write(f"{outcome}: {detail}\n")
          return {"accepted": 0, "rejected": 1, "skipped": 77}[outcome]


      if __name__ == "__main__":
          raise SystemExit(main())
      ```
